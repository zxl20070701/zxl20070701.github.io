Hapi路由和请求处理

基础路由

GET请求 

const Hapi = require('@hapi/hapi');

const init = async () => {
    const server = Hapi.server({
        port: 3000,
        host: 'localhost'
    });

    server.route({
        method: 'GET',
        path: '/users',
        handler: (request, h) => {
            return { users: [] };
        }
    });
};

其中request.query包含GET请求的查询参数。

POST请求 

server.route({
    method: 'POST',
    path: '/users',
    handler: (request, h) => {
        const body = request.payload;
        return { success: true };
    }
});

其中request.payload包含POST请求的请求体数据。

插件系统

解析JSON 

await server.register({
    plugin: require('@hapi/inert')
});

设置响应头 

server.route({
    method: 'GET',
    path: '/users',
    handler: (request, h) => {
        return h.response({ users: [] }).header('Content-Type', 'application/json; charset=utf-8');
    }
});

路由参数 

server.route({
    method: 'GET',
    path: '/users/{id}',
    handler: (request, h) => {
        const userId = request.params.id;
        return { id: userId };
    }
});

静态文件服务 

await server.register(require('@hapi/inert'));

server.route({
    method: 'GET',
    path: '/{param*}',
    handler: {
        directory: {
            path: 'public'
        }
    }
});

错误处理 

server.ext('onPreResponse', (request, h) => {
    const response = request.response;
    if (response.isBoom) {
        return h.response({ error: 'Something went wrong!' }).code(500);
    }
    return h.continue;
});

环境变量配置 

require('dotenv').config();
const port = process.env.PORT || 3000;

const server = Hapi.server({
    port: port,
    host: process.env.HOST || 'localhost'
});

路由分组 

const userRoutes = {
    name: 'user-routes',
    register: async function (server, options) {
        server.route([
            {
                method: 'GET',
                path: '/users',
                handler: (request, h) => {
                    return { users: [] };
                }
            },
            {
                method: 'GET',
                path: '/users/{id}',
                handler: (request, h) => {
                    return { id: request.params.id };
                }
            }
        ]);
    }
};

await server.register({
    plugin: userRoutes,
    routes: { prefix: '/api' }
});
