工作区信息

获取工作区文件夹 

const workspaceFolders = vscode.workspace.workspaceFolders;
if (workspaceFolders) {
    workspaceFolders.forEach(folder => {
        console.log('工作区:', folder.name, folder.uri.fsPath);
    });
}

监听工作区变化 

// 监听工作区文件夹变化
vscode.workspace.onDidChangeWorkspaceFolders(event => {
    console.log('工作区已更改');
    event.added.forEach(folder => console.log('添加:', folder.uri.fsPath));
    event.removed.forEach(folder => console.log('移除:', folder.uri.fsPath));
});

文件系统监听

创建文件监听器 

// 监听特定模式的文件变化
const watcher = vscode.workspace.createFileSystemWatcher('**/*.js');
watcher.onDidCreate(uri => console.log('文件创建:', uri.fsPath));
watcher.onDidChange(uri => console.log('文件修改:', uri.fsPath));
watcher.onDidDelete(uri => console.log('文件删除:', uri.fsPath));

查找文件

查找文件内容 

// 在工作区中查找文本
const files = await vscode.workspace.findFiles('**/*.js', '**/node_modules/**');
for (const file of files) {
    const document = await vscode.workspace.openTextDocument(file);
    const text = document.getText();
    if (text.includes('searchText')) {
        console.log('在文件中找到:', file.fsPath);
    }
}
