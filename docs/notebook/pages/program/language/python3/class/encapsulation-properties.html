封装与属性

封装的概念

封装（Encapsulation）是面向对象编程的核心原则之一，它将数据（属性）和操作数据的方法绑定在一起，并隐藏对象的内部实现细节，只暴露必要的接口给外部使用。

私有属性和方法

在Python中，通过在属性或方法名前添加双下划线__来定义私有成员： 

class BankAccount:
    def __init__(self, account_number, initial_balance):
        self.account_number = account_number  # 公有属性
        self.__balance = initial_balance      # 私有属性
        self.__transaction_history = []     # 私有属性
    
    def deposit(self, amount):
        """公有方法：存款"""
        if amount > 0:
            self.__balance += amount
            self.__transaction_history.append(f"Deposit: +${amount}")
            return True
        return False
    
    def withdraw(self, amount):
        """公有方法：取款"""
        if 0 < amount <= self.__balance:
            self.__balance -= amount
            self.__transaction_history.append(f"Withdrawal: -${amount}")
            return True
        return False
    
    def get_balance(self):
        """公有方法：获取余额"""
        return self.__balance
    
    def get_account_number(self):
        """公有方法：获取账号"""
        return self.account_number
    
    def __calculate_interest(self):
        """私有方法：计算利息"""
        return self.__balance * 0.05
    
    def apply_interest(self):
        """公有方法：应用利息"""
        interest = self.__calculate_interest()
        self.__balance += interest
        self.__transaction_history.append(f"Interest: +${interest:.2f}")

# 使用封装类
account = BankAccount("1234567890", 1000)

# 可以访问公有属性和方法
print(f"Account Number: {account.get_account_number()}")
print(f"Initial Balance: ${account.get_balance()}")

# 进行操作
account.deposit(500)
account.withdraw(200)
account.apply_interest()

print(f"Final Balance: ${account.get_balance()}")

# print(account.__balance)  # AttributeError
# print(account.__transaction_history)  # AttributeError
# print(account.__calculate_interest())  # AttributeError

属性的getter和setter

使用@property装饰器

@property装饰器用于创建属性的getter方法，可以控制对属性的访问： 

class Temperature:
    def __init__(self, celsius=0):
        self._celsius = celsius  # 受保护属性
    
    @property
    def celsius(self):
        """摄氏度属性的getter"""
        return self._celsius
    
    @celsius.setter
    def celsius(self, value):
        """摄氏度属性的setter"""
        if value < -273.15:
            raise ValueError("温度不能低于绝对零度(-273.15°C)")
        self._celsius = value
    
    @property
    def fahrenheit(self):
        """华氏度属性的getter（只读）"""
        return self._celsius * 9/5 + 32
    
    @fahrenheit.setter
    def fahrenheit(self, value):
        """华氏度属性的setter"""
        if value < -459.67:
            raise ValueError("温度不能低于绝对零度(-459.67°F)")
        self._celsius = (value - 32) * 5/9
    
    @property
    def kelvin(self):
        """开尔文温度的getter（只读）"""
        return self._celsius + 273.15

# 使用属性
temp = Temperature(25)

# 使用getter
print(f"{temp.celsius}°C = {temp.fahrenheit}°F = {temp.kelvin}K")

# 使用setter
temp.celsius = 30
print(f"After setting to 30°C: {temp.celsius}°C = {temp.fahrenheit}°F")

temp.fahrenheit = 86
print(f"After setting to 86°F: {temp.celsius}°C = {temp.fahrenheit}°F")

# 尝试设置无效温度
try:
    temp.celsius = -300
except ValueError as e:
    print(f"Error: {e}")

计算属性

使用@property创建计算属性，这些属性不存储实际数据，而是根据其他属性计算得出： 

class Rectangle:
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    @property
    def area(self):
        """计算面积（只读属性）"""
        return self.width * self.height
    
    @property
    def perimeter(self):
        """计算周长（只读属性）"""
        return 2 * (self.width + self.height)
    
    @property
    def diagonal(self):
        """计算对角线长度（只读属性）"""
        import math
        return math.sqrt(self.width**2 + self.height**2)
    
    def __str__(self):
        return f"Rectangle({self.width}x{self.height})"

# 使用计算属性
rect = Rectangle(3, 4)
print(rect)
print(f"Area: {rect.area}")
print(f"Perimeter: {rect.perimeter}")
print(f"Diagonal: {rect.diagonal:.2f}")

# 修改尺寸
rect.width = 5
rect.height = 12
print(f"After resizing: Area = {rect.area}, Perimeter = {rect.perimeter}")

属性验证

在setter中添加验证逻辑，确保数据的有效性： 

class Person:
    def __init__(self, name, age, email):
        self.name = name
        self.age = age
        self.email = email
    
    @property
    def name(self):
        return self._name
    
    @name.setter
    def name(self, value):
        if not value or not value.strip():
            raise ValueError("姓名不能为空")
        self._name = value.strip()
    
    @property
    def age(self):
        return self._age
    
    @age.setter
    def age(self, value):
        if not isinstance(value, int) or value < 0 or value > 150:
            raise ValueError("年龄必须是0-150之间的整数")
        self._age = value
    
    @property
    def email(self):
        return self._email
    
    @email.setter
    def email(self, value):
        if "@" not in value or "." not in value.split("@")[1]:
            raise ValueError("无效的邮箱格式")
        self._email = value
    
    def __str__(self):
        return f"Person(name='{self.name}', age={self.age}, email='{self.email}')"

# 使用带验证的属性
try:
    person = Person("Alice", 30, "alice@example.com")
    print(person)
    
    # 修改属性
    person.name = "Alice Smith"
    person.age = 31
    person.email = "alice.smith@example.com"
    print(person)
    
    # 尝试设置无效值
    person.age = -5  # 会抛出异常
except ValueError as e:
    print(f"Error: {e}")

只读属性

创建只读属性，只允许获取值而不允许设置： 

class Circle:
    def __init__(self, radius):
        self.radius = radius  # 使用setter进行验证
    
    @property
    def radius(self):
        return self._radius
    
    @radius.setter
    def radius(self, value):
        if value <= 0:
            raise ValueError("半径必须大于0")
        self._radius = value
    
    @property
    def diameter(self):
        """直径（只读）"""
        return self._radius * 2
    
    @property
    def area(self):
        """面积（只读）"""
        import math
        return math.pi * self._radius ** 2
    
    @property
    def circumference(self):
        """周长（只读）"""
        import math
        return 2 * math.pi * self._radius
    
    def __str__(self):
        return f"Circle(radius={self.radius})"

# 使用只读属性
circle = Circle(5)
print(circle)
print(f"Radius: {circle.radius}")
print(f"Diameter: {circle.diameter}")
print(f"Area: {circle.area:.2f}")
print(f"Circumference: {circle.circumference:.2f}")

# 可以修改radius
circle.radius = 10
print(f"After changing radius to 10: Area = {circle.area:.2f}")

# 但不能直接修改只读属性
try:
    circle.diameter = 20  # AttributeError
except AttributeError as e:
    print(f"Error: {e}")

属性缓存

使用缓存提高计算密集型属性的性能： 

class DataProcessor:
    def __init__(self, data):
        self.data = data
        self._cached_stats = None
    
    @property
    def stats(self):
        """计算并缓存统计数据"""
        if self._cached_stats is None:
            print("Calculating statistics...")
            self._cached_stats = {
                'count': len(self.data),
                'sum': sum(self.data),
                'mean': sum(self.data) / len(self.data),
                'min': min(self.data),
                'max': max(self.data)
            }
        return self._cached_stats
    
    def add_data(self, new_data):
        """添加新数据并清除缓存"""
        self.data.extend(new_data)
        self._cached_stats = None  # 清除缓存
    
    def clear_cache(self):
        """手动清除缓存"""
        self._cached_stats = None

# 使用缓存属性
data = [1, 2, 3, 4, 5]
processor = DataProcessor(data)

# 第一次访问会计算
print("First access:")
print(processor.stats)

# 第二次访问会使用缓存
print("\nSecond access:")
print(processor.stats)

# 添加新数据后缓存会被清除
processor.add_data([6, 7, 8])
print("\nAfter adding data:")
print(processor.stats)

属性描述符

使用描述符创建可重用的属性验证逻辑： 

class ValidatedAttribute:
    """属性描述符基类"""
    def __init__(self, name):
        self.name = name
    
    def __get__(self, instance, owner):
        if instance is None:
            return self
        return instance.__dict__[self.name]
    
    def __set__(self, instance, value):
        self.validate(value)
        instance.__dict__[self.name] = value
    
    def __delete__(self, instance):
        del instance.__dict__[self.name]
    
    def validate(self, value):
        """验证方法，子类需要实现"""
        pass

class PositiveNumber(ValidatedAttribute):
    """正数验证描述符"""
    def validate(self, value):
        if not isinstance(value, (int, float)) or value <= 0:
            raise ValueError(f"{self.name} must be a positive number")

class StringNotEmpty(ValidatedAttribute):
    """非空字符串验证描述符"""
    def validate(self, value):
        if not isinstance(value, str) or not value.strip():
            raise ValueError(f"{self.name} must be a non-empty string")

class Person:
    name = StringNotEmpty('name')
    age = PositiveNumber('age')
    height = PositiveNumber('height')
    
    def __init__(self, name, age, height):
        self.name = name
        self.age = age
        self.height = height
    
    def __str__(self):
        return f"Person(name='{self.name}', age={self.age}, height={self.height})"

# 使用描述符验证属性
try:
    person = Person("Alice", 30, 165)
    print(person)
    
    # 有效修改
    person.age = 31
    print(person)
    
    # 无效修改
    person.age = -5  # 会抛出异常
except ValueError as e:
    print(f"Error: {e}")
