抽象类与接口

抽象类与接口概述

抽象类（Abstract Class）是不能被实例化的类，它定义了一组抽象方法，子类必须实现这些方法。接口（Interface）是一种特殊的抽象类，它只定义方法签名而不提供实现。

使用abc模块创建抽象类

Python的abc模块提供了创建抽象类的工具： 

from abc import ABC, abstractmethod

class Shape(ABC):
    """抽象基类：图形"""
    
    @abstractmethod
    def area(self):
        """计算面积的抽象方法"""
        pass
    
    @abstractmethod
    def perimeter(self):
        """计算周长的抽象方法"""
        pass
    
    def describe(self):
        """具体方法，所有子类都可以使用"""
        return f"This is a {self.__class__.__name__}"

# 尝试实例化抽象类会报错
# shape = Shape()  # TypeError: Can't instantiate abstract class Shape

class Rectangle(Shape):
    """矩形类，继承抽象基类"""
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

class Circle(Shape):
    """圆形类，继承抽象基类"""
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        import math
        return math.pi * self.radius ** 2
    
    def perimeter(self):
        import math
        return 2 * math.pi * self.radius

# 使用抽象类的子类
rectangle = Rectangle(5, 3)
circle = Circle(4)

print(rectangle.describe())  # This is a Rectangle
print(f"Rectangle area: {rectangle.area()}")       # 15
print(f"Rectangle perimeter: {rectangle.perimeter()}")  # 16

print(circle.describe())     # This is a Circle
print(f"Circle area: {circle.area():.2f}")         # 50.27
print(f"Circle perimeter: {circle.perimeter():.2f}")    # 25.13

抽象属性

使用@property和@abstractmethod创建抽象属性： 

from abc import ABC, abstractmethod

class Vehicle(ABC):
    """抽象基类：交通工具"""
    
    @property
    @abstractmethod
    def max_speed(self):
        """抽象属性：最大速度"""
        pass
    
    @property
    @abstractmethod
    def fuel_type(self):
        """抽象属性：燃料类型"""
        pass
    
    @abstractmethod
    def start_engine(self):
        """抽象方法：启动引擎"""
        pass
    
    def display_info(self):
        """具体方法：显示信息"""
        return f"{self.__class__.__name__}: Max speed {self.max_speed} km/h, Fuel: {self.fuel_type}"

class Car(Vehicle):
    """汽车类"""
    @property
    def max_speed(self):
        return 200
    
    @property
    def fuel_type(self):
        return "Gasoline"
    
    def start_engine(self):
        return "Car engine started with key"

class ElectricCar(Vehicle):
    """电动汽车类"""
    @property
    def max_speed(self):
        return 180
    
    @property
    def fuel_type(self):
        return "Electricity"
    
    def start_engine(self):
        return "Electric motor started silently"

class Bicycle(Vehicle):
    """自行车类"""
    @property
    def max_speed(self):
        return 30
    
    @property
    def fuel_type(self):
        return "Human Power"
    
    def start_engine(self):
        return "Start pedaling!"

# 使用抽象属性的子类
vehicles = [Car(), ElectricCar(), Bicycle()]

for vehicle in vehicles:
    print(vehicle.display_info())
    print(vehicle.start_engine())
    print("-" * 40)

虚拟子类

使用register()方法创建虚拟子类，不需要实际继承： 

from abc import ABC, abstractmethod

class Drawable(ABC):
    """可绘制对象的抽象基类"""
    
    @abstractmethod
    def draw(self):
        pass
    
    @abstractmethod
    def get_bounds(self):
        pass

class SVGRenderer:
    """SVG渲染器（不是Drawable的子类）"""
    def __init__(self, content):
        self.content = content
    
    def draw(self):
        return f"Rendering SVG: {self.content}"
    
    def get_bounds(self):
        return (0, 0, 100, 100)

class CanvasRenderer:
    """Canvas渲染器（不是Drawable的子类）"""
    def __init__(self, content):
        self.content = content
    
    def draw(self):
        return f"Drawing on canvas: {self.content}"
    
    def get_bounds(self):
        return (10, 10, 200, 200)

# 注册虚拟子类
Drawable.register(SVGRenderer)
Drawable.register(CanvasRenderer)

# 现在这些类被认为是Drawable的子类
print(issubclass(SVGRenderer, Drawable))    # True
print(issubclass(CanvasRenderer, Drawable)) # True

# 可以创建实例并调用抽象方法
svg = SVGRenderer("circle")
canvas = CanvasRenderer("rectangle")

drawables = [svg, canvas]
for drawable in drawables:
    print(drawable.draw())
    print(f"Bounds: {drawable.get_bounds()}")
    print(isinstance(drawable, Drawable))  # True
    print("-" * 30)

接口定义与实现

创建纯接口类，只定义方法签名： 

from abc import ABC, abstractmethod

class DatabaseInterface(ABC):
    """数据库接口"""
    
    @abstractmethod
    def connect(self):
        """连接数据库"""
        pass
    
    @abstractmethod
    def disconnect(self):
        """断开连接"""
        pass
    
    @abstractmethod
    def execute_query(self, query):
        """执行查询"""
        pass
    
    @abstractmethod
    def insert_data(self, table, data):
        """插入数据"""
        pass
    
    @abstractmethod
    def update_data(self, table, data, condition):
        """更新数据"""
        pass
    
    @abstractmethod
    def delete_data(self, table, condition):
        """删除数据"""
        pass

class MySQLDatabase(DatabaseInterface):
    """MySQL数据库实现"""
    def connect(self):
        return "Connected to MySQL database"
    
    def disconnect(self):
        return "Disconnected from MySQL database"
    
    def execute_query(self, query):
        return f"Executing MySQL query: {query}"
    
    def insert_data(self, table, data):
        return f"Inserting data into MySQL table {table}: {data}"
    
    def update_data(self, table, data, condition):
        return f"Updating MySQL table {table} with {data} where {condition}"
    
    def delete_data(self, table, condition):
        return f"Deleting from MySQL table {table} where {condition}"

class PostgreSQLDatabase(DatabaseInterface):
    """PostgreSQL数据库实现"""
    def connect(self):
        return "Connected to PostgreSQL database"
    
    def disconnect(self):
        return "Disconnected from PostgreSQL database"
    
    def execute_query(self, query):
        return f"Executing PostgreSQL query: {query}"
    
    def insert_data(self, table, data):
        return f"Inserting data into PostgreSQL table {table}: {data}"
    
    def update_data(self, table, data, condition):
        return f"Updating PostgreSQL table {table} with {data} where {condition}"
    
    def delete_data(self, table, condition):
        return f"Deleting from PostgreSQL table {table} where {condition}"

# 使用接口
databases = [MySQLDatabase(), PostgreSQLDatabase()]

for db in databases:
    print(db.connect())
    print(db.execute_query("SELECT * FROM users"))
    print(db.insert_data("users", {"name": "Alice", "age": 30}))
    print(db.disconnect())
    print("-" * 40)

混合抽象类

创建包含具体方法和抽象方法的混合抽象类： 

from abc import ABC, abstractmethod
from datetime import datetime

class Logger(ABC):
    """日志抽象类"""
    
    def __init__(self, name):
        self.name = name
    
    @abstractmethod
    def write_log(self, level, message):
        """抽象方法：写入日志"""
        pass
    
    def info(self, message):
        """具体方法：记录信息日志"""
        self.write_log("INFO", message)
    
    def warning(self, message):
        """具体方法：记录警告日志"""
        self.write_log("WARNING", message)
    
    def error(self, message):
        """具体方法：记录错误日志"""
        self.write_log("ERROR", message)
    
    def _format_message(self, level, message):
        """受保护方法：格式化日志消息"""
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        return f"[{timestamp}] [{level}] {self.name}: {message}"

class FileLogger(Logger):
    """文件日志实现"""
    def __init__(self, name, filename):
        super().__init__(name)
        self.filename = filename
    
    def write_log(self, level, message):
        formatted = self._format_message(level, message)
        with open(self.filename, "a") as f:
            f.write(formatted + "\n")
        print(f"Logged to file: {formatted}")

class ConsoleLogger(Logger):
    """控制台日志实现"""
    def write_log(self, level, message):
        formatted = self._format_message(level, message)
        print(f"Console log: {formatted}")

# 使用混合抽象类
file_logger = FileLogger("AppLogger", "app.log")
console_logger = ConsoleLogger("AppLogger")

loggers = [file_logger, console_logger]

for logger in loggers:
    logger.info("Application started")
    logger.warning("This is a warning")
    logger.error("An error occurred")
    print("-" * 40)

抽象工厂模式

使用抽象类实现抽象工厂模式： 

from abc import ABC, abstractmethod

# 抽象产品
class Button(ABC):
    @abstractmethod
    def render(self):
        pass
    
    @abstractmethod
    def on_click(self):
        pass

class Checkbox(ABC):
    @abstractmethod
    def render(self):
        pass
    
    @abstractmethod
    def toggle(self):
        pass

# 具体产品 - Windows风格
class WindowsButton(Button):
    def render(self):
        return "Rendering Windows-style button"
    
    def on_click(self):
        return "Windows button clicked"

class WindowsCheckbox(Checkbox):
    def render(self):
        return "Rendering Windows-style checkbox"
    
    def toggle(self):
        return "Windows checkbox toggled"

# 具体产品 - Mac风格
class MacButton(Button):
    def render(self):
        return "Rendering Mac-style button"
    
    def on_click(self):
        return "Mac button clicked"

class MacCheckbox(Checkbox):
    def render(self):
        return "Rendering Mac-style checkbox"
    
    def toggle(self):
        return "Mac checkbox toggled"

# 抽象工厂
class GUIFactory(ABC):
    @abstractmethod
    def create_button(self):
        pass
    
    @abstractmethod
    def create_checkbox(self):
        pass

# 具体工厂
class WindowsFactory(GUIFactory):
    def create_button(self):
        return WindowsButton()
    
    def create_checkbox(self):
        return WindowsCheckbox()

class MacFactory(GUIFactory):
    def create_button(self):
        return MacButton()
    
    def create_checkbox(self):
        return MacCheckbox()

# 使用抽象工厂
def create_ui(factory):
    """使用工厂创建UI组件"""
    button = factory.create_button()
    checkbox = factory.create_checkbox()
    
    print(button.render())
    print(button.on_click())
    print(checkbox.render())
    print(checkbox.toggle())

# 创建不同风格的UI
windows_factory = WindowsFactory()
mac_factory = MacFactory()

print("=== Windows UI ===")
create_ui(windows_factory)

print("\n=== Mac UI ===")
create_ui(mac_factory)

抽象基类的实用工具

使用ABC模块提供的实用工具函数： 

from abc import ABC, abstractmethod

class Animal(ABC):
    @abstractmethod
    def make_sound(self):
        pass

class Dog(Animal):
    def make_sound(self):
        return "Woof!"

class Cat:
    def make_sound(self):
        return "Meow!"
    
    def purr(self):
        return "Purr..."

# 检查类是否为抽象类
print(Animal.__abstractmethods__)  # {'make_sound'}
print(Dog.__abstractmethods__)     # set()

# 使用register注册虚拟子类
Animal.register(Cat)
print(issubclass(Cat, Animal))     # True
print(isinstance(Cat(), Animal))   # True

# 获取类的所有抽象方法
def get_abstract_methods(cls):
    """获取类的所有抽象方法"""
    if not hasattr(cls, '__abstractmethods__'):
        return set()
    return cls.__abstractmethods__

print(get_abstract_methods(Animal))  # {'make_sound'}
print(get_abstract_methods(Dog))      # set()

# 检查类是否实现了所有抽象方法
def is_concrete_class(cls):
    """检查类是否为具体类（非抽象类）"""
    return len(get_abstract_methods(cls)) == 0

print(is_concrete_class(Animal))  # False
print(is_concrete_class(Dog))       # True
print(is_concrete_class(Cat))       # True
