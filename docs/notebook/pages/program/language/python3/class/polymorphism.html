多态性

多态概念

多态（Polymorphism）意味着"多种形态"，允许不同类的对象对同一消息做出不同的响应。多态性使得代码更加灵活和可扩展。 

class Shape:
    def area(self):
        raise NotImplementedError("子类必须实现area方法")
    
    def perimeter(self):
        raise NotImplementedError("子类必须实现perimeter方法")

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        import math
        return math.pi * self.radius ** 2
    
    def perimeter(self):
        import math
        return 2 * math.pi * self.radius

class Triangle(Shape):
    def __init__(self, a, b, c):
        self.a, self.b, self.c = a, b, c
    
    def area(self):
        # 使用海伦公式
        import math
        s = (self.a + self.b + self.c) / 2
        return math.sqrt(s * (s - self.a) * (s - self.b) * (s - self.c))
    
    def perimeter(self):
        return self.a + self.b + self.c

# 多态性演示
shapes = [
    Rectangle(4, 5),
    Circle(3),
    Triangle(3, 4, 5)
]

for shape in shapes:
    print(f"面积: {shape.area():.2f}, 周长: {shape.perimeter():.2f}")

鸭子类型（Duck Typing） 

class Duck:
    def quack(self):
        return "Quack! Quack!"
    
    def fly(self):
        return "Flying like a duck!"

class Person:
    def quack(self):
        return "I'm quacking like a duck!"
    
    def fly(self):
        return "I'm flying like an airplane!"

def make_it_quack_and_fly(thing):
    """只要对象有quack和fly方法就可以调用"""
    print(thing.quack())
    print(thing.fly())

# 不同类型的对象都可以使用
duck = Duck()
person = Person()

make_it_quack_and_fly(duck)
make_it_quack_and_fly(person)

运算符重载实现多态 

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __add__(self, other):
        """重载+运算符"""
        return Vector(self.x + other.x, self.y + other.y)
    
    def __str__(self):
        return f"Vector({self.x}, {self.y})"
    
    def __eq__(self, other):
        """重载==运算符"""
        return self.x == other.x and self.y == other.y

# 使用重载的运算符
v1 = Vector(2, 3)
v2 = Vector(4, 5)
v3 = v1 + v2

print(v3)  # Vector(6, 8)
print(v1 == Vector(2, 3))  # True
