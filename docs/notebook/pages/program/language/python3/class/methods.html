实例方法、类方法与静态方法

实例方法

实例方法是最常见的方法类型，第一个参数必须是self，表示对象本身。实例方法可以访问和修改对象的属性。 

class Car:
    def __init__(self, brand, model, year):
        self.brand = brand
        self.model = model
        self.year = year
        self.mileage = 0
    
    def drive(self, miles):
        """实例方法：增加里程"""
        self.mileage += miles
        return f"{self.brand} {self.model} 行驶了 {miles} 英里"
    
    def get_info(self):
        """实例方法：获取汽车信息"""
        return f"{self.year} {self.brand} {self.model}, 里程: {self.mileage} 英里"

# 创建对象并调用实例方法
my_car = Car("Toyota", "Camry", 2020)
print(my_car.drive(100))      # Toyota Camry 行驶了 100 英里
print(my_car.get_info())      # 2020 Toyota Camry, 里程: 100 英里

类方法

类方法使用@classmethod装饰器定义，第一个参数必须是cls，表示类本身。类方法可以访问和修改类属性，但不能直接访问实例属性。 

class Student:
    # 类属性
    total_students = 0
    school_name = "Python University"
    
    def __init__(self, name, grade):
        self.name = name
        self.grade = grade
        # 每创建一个学生，总数加1
        Student.total_students += 1
    
    @classmethod
    def get_total_students(cls):
        """类方法：获取学生总数"""
        return cls.total_students
    
    @classmethod
    def set_school_name(cls, name):
        """类方法：设置学校名称"""
        cls.school_name = name
    
    @classmethod
    def create_from_string(cls, student_string):
        """类方法：从字符串创建学生对象"""
        name, grade = student_string.split('-')
        return cls(name, grade)

# 使用类方法
student1 = Student("Alice", "A")
student2 = Student("Bob", "B")

print(Student.get_total_students())  # 输出: 2
print(Student.school_name)           # 输出: Python University

# 修改类属性
Student.set_school_name("Advanced Python University")
print(Student.school_name)           # 输出: Advanced Python University

# 使用类方法创建对象
student3 = Student.create_from_string("Charlie-C")
print(student3.name)  # 输出: Charlie
print(student3.grade) # 输出: C

静态方法

静态方法使用@staticmethod装饰器定义，不需要self或cls参数。静态方法不能访问类属性或实例属性，它们只是组织在类中的普通函数。 

class MathUtils:
    @staticmethod
    def add(x, y):
        """静态方法：加法运算"""
        return x + y
    
    @staticmethod
    def multiply(x, y):
        """静态方法：乘法运算"""
        return x * y
    
    @staticmethod
    def is_even(number):
        """静态方法：判断是否为偶数"""
        return number % 2 == 0

# 调用静态方法
print(MathUtils.add(5, 3))        # 输出: 8
print(MathUtils.multiply(4, 7))   # 输出: 28
print(MathUtils.is_even(10))      # 输出: True

# 也可以通过实例调用静态方法
utils = MathUtils()
print(utils.add(2, 3))  # 输出: 5

三种方法的对比 

class Weather:
    # 类属性
    default_city = "Beijing"
    
    def __init__(self, city=None):
        # 实例属性
        self.city = city or self.default_city
        self.temperature = None
    
    # 实例方法 - 可以访问实例属性和类属性
    def set_temperature(self, temp):
        self.temperature = temp
    
    def get_info(self):
        return f"{self.city}: {self.temperature}°C"
    
    # 类方法 - 可以访问类属性，但不能直接访问实例属性
    @classmethod
    def set_default_city(cls, city):
        cls.default_city = city
    
    @classmethod
    def create_default_weather(cls):
        return cls()  # 使用类属性创建实例
    
    # 静态方法 - 不能访问类属性或实例属性
    @staticmethod
    def celsius_to_fahrenheit(celsius):
        return celsius * 9/5 + 32
    
    @staticmethod
    def is_extreme_temperature(temp):
        return temp < -20 or temp > 40

# 演示三种方法的使用
weather = Weather("Shanghai")
weather.set_temperature(25)
print(weather.get_info())  # Shanghai: 25°C

# 使用类方法
Weather.set_default_city("Guangzhou")
default_weather = Weather.create_default_weather()
print(default_weather.city)  # Guangzhou

# 使用静态方法
print(Weather.celsius_to_fahrenheit(25))  # 77.0
print(Weather.is_extreme_temperature(45)) # True

实际应用示例

一个综合示例，展示三种方法在实际中的应用： 

class DatabaseConnection:
    # 类属性 - 共享的连接池
    connection_pool = []
    max_connections = 5
    
    def __init__(self, host, port, database):
        self.host = host
        self.port = port
        self.database = database
        self.is_connected = False
    
    # 实例方法 - 操作特定连接
    def connect(self):
        if not self.is_connected:
            print(f"连接到 {self.host}:{self.port}/{self.database}")
            self.is_connected = True
    
    def disconnect(self):
        if self.is_connected:
            print(f"断开连接 {self.host}:{self.port}/{self.database}")
            self.is_connected = False
    
    # 类方法 - 管理连接池
    @classmethod
    def get_connection_count(cls):
        return len(cls.connection_pool)
    
    @classmethod
    def add_to_pool(cls, connection):
        if len(cls.connection_pool) < cls.max_connections:
            cls.connection_pool.append(connection)
            return True
        return False
    
    # 静态方法 - 工具函数
    @staticmethod
    def validate_connection_string(conn_string):
        """验证连接字符串格式"""
        parts = conn_string.split(':')
        return len(parts) == 3 and parts[0] and parts[1].isdigit() and parts[2]

# 使用示例
# 验证连接字符串
is_valid = DatabaseConnection.validate_connection_string("localhost:5432:mydb")
print(f"连接字符串有效: {is_valid}")  # True

# 创建连接
conn1 = DatabaseConnection("localhost", 5432, "mydb")
conn2 = DatabaseConnection("remotehost", 3306, "webapp")

# 使用实例方法
conn1.connect()
conn2.connect()

# 使用类方法管理连接池
DatabaseConnection.add_to_pool(conn1)
DatabaseConnection.add_to_pool(conn2)
print(f"连接池中的连接数: {DatabaseConnection.get_connection_count()}")  # 2

# 断开连接
conn1.disconnect()
conn2.disconnect()
