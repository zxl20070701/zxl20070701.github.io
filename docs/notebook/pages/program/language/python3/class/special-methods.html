特殊方法与运算符重载

特殊方法概述

特殊方法（Special Methods），也称为魔术方法（Magic Methods）或双下划线方法（Dunder Methods），是Python中以双下划线开头和结尾的方法，如__init__、__str__等。这些方法允许我们自定义类的行为，实现运算符重载和协议实现。

对象创建与销毁

__new__方法

__new__是创建对象时调用的第一个方法，负责创建实例： 

class Singleton:
    """单例模式实现"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        if not hasattr(self, 'initialized'):
            self.initialized = True
            self.value = 0

# 测试单例
s1 = Singleton()
s2 = Singleton()
print(s1 is s2)  # True，同一个实例
s1.value = 100
print(s2.value)  # 100

__del__方法

__del__是析构方法，在对象被销毁时调用： 

class FileHandler:
    def __init__(self, filename):
        self.filename = filename
        self.file = open(filename, 'w')
        print(f"Opened file: {filename}")
    
    def __del__(self):
        if hasattr(self, 'file') and self.file:
            self.file.close()
            print(f"Closed file: {self.filename}")

# 使用文件处理器
handler = FileHandler("test.txt")
# 当handler被销毁时，__del__会被调用

字符串表示

__str__方法

__str__定义对象的"非正式"字符串表示，用于print()和str()： 

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):
        return f"Point({self.x}, {self.y})"

p = Point(3, 4)
print(p)        # Point(3, 4)
print(str(p))   # Point(3, 4)

__repr__方法

__repr__定义对象的"官方"字符串表示，用于repr()和调试： 

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):
        return f"Point({self.x}, {self.y})"
    
    def __repr__(self):
        return f"Point(x={self.x}, y={self.y})"

p = Point(3, 4)
print(str(p))   # Point(3, 4)
print(repr(p))  # Point(x=3, y=4)

比较运算符重载 

class Student:
    def __init__(self, name, score):
        self.name = name
        self.score = score
    
    def __eq__(self, other):
        """等于=="""
        if isinstance(other, Student):
            return self.score == other.score
        return False
    
    def __lt__(self, other):
        """小于<"""
        if isinstance(other, Student):
            return self.score < other.score
        return NotImplemented
    
    def __le__(self, other):
        """小于等于<="""
        if isinstance(other, Student):
            return self.score <= other.score
        return NotImplemented
    
    def __gt__(self, other):
        """大于>"""
        if isinstance(other, Student):
            return self.score > other.score
        return NotImplemented
    
    def __ge__(self, other):
        """大于等于>="""
        if isinstance(other, Student):
            return self.score >= other.score
        return NotImplemented
    
    def __ne__(self, other):
        """不等于!="""
        return not self.__eq__(other)
    
    def __str__(self):
        return f"{self.name}: {self.score}"

# 使用比较运算符
alice = Student("Alice", 85)
bob = Student("Bob", 90)
charlie = Student("Charlie", 85)

print(alice < bob)    # True
print(alice <= bob)   # True
print(alice > bob)    # False
print(alice >= bob)   # False
print(alice == charlie)  # True
print(alice != bob)   # True

算术运算符重载 

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __add__(self, other):
        """加法+"""
        if isinstance(other, Vector):
            return Vector(self.x + other.x, self.y + other.y)
        return NotImplemented
    
    def __sub__(self, other):
        """减法-"""
        if isinstance(other, Vector):
            return Vector(self.x - other.x, self.y - other.y)
        return NotImplemented
    
    def __mul__(self, scalar):
        """乘法*（向量与标量）"""
        if isinstance(scalar, (int, float)):
            return Vector(self.x * scalar, self.y * scalar)
        return NotImplemented
    
    def __rmul__(self, scalar):
        """右乘法*（标量与向量）"""
        return self.__mul__(scalar)
    
    def __truediv__(self, scalar):
        """除法/"""
        if isinstance(scalar, (int, float)) and scalar != 0:
            return Vector(self.x / scalar, self.y / scalar)
        return NotImplemented
    
    def __neg__(self):
        """负号-"""
        return Vector(-self.x, -self.y)
    
    def __str__(self):
        return f"Vector({self.x}, {self.y})"
    
    def __repr__(self):
        return f"Vector({self.x}, {self.y})"

# 使用算术运算符
v1 = Vector(2, 3)
v2 = Vector(4, 1)

print(v1 + v2)    # Vector(6, 4)
print(v1 - v2)    # Vector(-2, 2)
print(v1 * 3)     # Vector(6, 9)
print(2 * v1)     # Vector(4, 6)
print(v1 / 2)     # Vector(1.0, 1.5)
print(-v1)        # Vector(-2, -3)

容器类型协议

__len__和__getitem__ 

class MyList:
    def __init__(self, items=None):
        self.items = items or []
    
    def __len__(self):
        """返回容器长度"""
        return len(self.items)
    
    def __getitem__(self, index):
        """获取指定索引的元素"""
        return self.items[index]
    
    def __setitem__(self, index, value):
        """设置指定索引的元素"""
        self.items[index] = value
    
    def __delitem__(self, index):
        """删除指定索引的元素"""
        del self.items[index]
    
    def __contains__(self, item):
        """检查是否包含元素"""
        return item in self.items
    
    def __iter__(self):
        """返回迭代器"""
        return iter(self.items)

# 使用容器协议
my_list = MyList([1, 2, 3, 4, 5])
print(len(my_list))        # 5
print(my_list[2])          # 3
my_list[1] = 10
print(my_list[1])          # 10
print(3 in my_list)        # True
del my_list[0]
print(len(my_list))        # 4

for item in my_list:
    print(item, end=' ')   # 10 3 4 5

上下文管理器协议

__enter__和__exit__方法 

class Timer:
    def __init__(self, name="Timer"):
        self.name = name
    
    def __enter__(self):
        """进入上下文时调用"""
        import time
        self.start_time = time.time()
        print(f"{self.name} started")
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出上下文时调用"""
        import time
        end_time = time.time()
        elapsed = end_time - self.start_time
        print(f"{self.name} finished in {elapsed:.4f} seconds")
        return False  # 不抑制异常

# 使用上下文管理器
with Timer("Loop Timer"):
    total = 0
    for i in range(1000000):
        total += i
    print(f"Sum: {total}")

# 自定义文件处理器
class FileHandler:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
    
    def __enter__(self):
        self.file = open(self.filename, self.mode)
        return self.file
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.file.close()
        if exc_type is not None:
            print(f"An error occurred: {exc_val}")
        return False

# 使用自定义上下文管理器
with FileHandler("test.txt", "w") as f:
    f.write("Hello, World!")

可调用对象协议

__call__方法 

class Counter:
    def __init__(self):
        self.count = 0
    
    def __call__(self, increment=1):
        """使对象可调用"""
        self.count += increment
        return self.count
    
    def reset(self):
        self.count = 0

# 使用可调用对象
counter = Counter()
print(counter())     # 1
print(counter())     # 2
print(counter(5))    # 7
counter.reset()
print(counter())     # 1

# 更复杂的可调用对象
class Polynomial:
    def __init__(self, *coefficients):
        """coefficients: a, b, c, ... for ax^n + bx^(n-1) + ... + c"""
        self.coefficients = coefficients
    
    def __call__(self, x):
        """计算多项式的值"""
        result = 0
        for power, coeff in enumerate(reversed(self.coefficients)):
            result += coeff * (x ** power)
        return result
    
    def __str__(self):
        terms = []
        for power, coeff in enumerate(reversed(self.coefficients)):
            if coeff != 0:
                if power == 0:
                    terms.append(str(coeff))
                elif power == 1:
                    terms.append(f"{coeff}x")
                else:
                    terms.append(f"{coeff}x^{power}")
        return " + ".join(terms)

# 创建多项式 f(x) = 2x^2 + 3x + 1
poly = Polynomial(2, 3, 1)
print(poly)                    # 2x^2 + 3x + 1
print(poly(2))                 # 15
print(poly(0))                 # 1
print(poly(-1))                # 0

数值类型协议

数值转换方法 

class Fraction:
    def __init__(self, numerator, denominator):
        if denominator == 0:
            raise ValueError("Denominator cannot be zero")
        self.numerator = numerator
        self.denominator = denominator
        self._simplify()
    
    def _simplify(self):
        """简化分数"""
        import math
        gcd = math.gcd(abs(self.numerator), abs(self.denominator))
        self.numerator //= gcd
        self.denominator //= gcd
        if self.denominator < 0:
            self.numerator = -self.numerator
            self.denominator = -self.denominator
    
    def __int__(self):
        """转换为整数"""
        return self.numerator // self.denominator
    
    def __float__(self):
        """转换为浮点数"""
        return self.numerator / self.denominator
    
    def __bool__(self):
        """转换为布尔值"""
        return self.numerator != 0
    
    def __abs__(self):
        """绝对值"""
        return Fraction(abs(self.numerator), self.denominator)
    
    def __str__(self):
        if self.denominator == 1:
            return str(self.numerator)
        return f"{self.numerator}/{self.denominator}"
    
    def __repr__(self):
        return f"Fraction({self.numerator}, {self.denominator})"

# 使用数值转换
f1 = Fraction(3, 4)
f2 = Fraction(6, 8)  # 会自动简化为3/4

print(int(f1))     # 0
print(float(f1))   # 0.75
print(bool(f1))    # True
print(abs(f1))     # 3/4

f3 = Fraction(0, 5)
print(bool(f3))    # False

print(f1)          # 3/4
print(repr(f1))    # Fraction(3, 4)

集合协议

__hash__和__eq__ 

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __eq__(self, other):
        """相等性比较"""
        if isinstance(other, Point):
            return self.x == other.x and self.y == other.y
        return False
    
    def __hash__(self):
        """计算哈希值，使对象可用于集合和字典键"""
        return hash((self.x, self.y))
    
    def __str__(self):
        return f"Point({self.x}, {self.y})"

# 使用集合协议
p1 = Point(1, 2)
p2 = Point(1, 2)
p3 = Point(3, 4)

print(p1 == p2)     # True
print(p1 == p3)     # False

# 可以放入集合中
point_set = {p1, p2, p3}
print(len(point_set))  # 2，因为p1和p2相等

# 可以作为字典键
point_dict = {p1: "First point", p3: "Third point"}
print(point_dict[p2])  # First point，因为p1和p2相等
