继承与多态

继承基础

继承（Inheritance）是面向对象编程的重要特性，允许一个类（子类）继承另一个类（父类）的属性和方法。子类可以重用父类的代码，并可以添加新的功能或重写父类的方法。 

class Animal:
    """父类：动物"""
    def __init__(self, name, species):
        self.name = name
        self.species = species
    
    def make_sound(self):
        return "Some generic animal sound"
    
    def eat(self):
        return f"{self.name} is eating"
    
    def sleep(self):
        return f"{self.name} is sleeping"

class Dog(Animal):
    """子类：狗，继承自动物"""
    def __init__(self, name, breed):
        # 调用父类的构造函数
        super().__init__(name, "Dog")
        self.breed = breed
    
    # 重写父类的方法
    def make_sound(self):
        return "Woof! Woof!"
    
    # 添加新的方法
    def fetch(self):
        return f"{self.name} is fetching the ball"

class Cat(Animal):
    """子类：猫，继承自动物"""
    def __init__(self, name, color):
        super().__init__(name, "Cat")
        self.color = color
    
    def make_sound(self):
        return "Meow! Meow!"
    
    def climb(self):
        return f"{self.name} is climbing a tree"

# 使用继承的类
dog = Dog("Buddy", "Golden Retriever")
cat = Cat("Whiskers", "Orange")

print(f"{dog.name} is a {dog.breed} {dog.species}")
print(f"{cat.name} is a {cat.color} {cat.species}")

print(dog.make_sound())  # Woof! Woof!
print(cat.make_sound())  # Meow! Meow!

print(dog.eat())    # 继承自父类
print(cat.sleep())  # 继承自父类

print(dog.fetch())  # Dog类特有的方法
print(cat.climb())  # Cat类特有的方法

多重继承

Python支持多重继承，一个类可以继承多个父类： 

class Flyable:
    """飞行能力"""
    def fly(self):
        return "I can fly!"

class Swimmable:
    """游泳能力"""
    def swim(self):
        return "I can swim!"

class Walkable:
    """行走能力"""
    def walk(self):
        return "I can walk!"

class Duck(Flyable, Swimmable, Walkable):
    """鸭子：会飞、会游泳、会走"""
    def __init__(self, name):
        self.name = name
    
    def quack(self):
        return "Quack! Quack!"

class Penguin(Swimmable, Walkable):
    """企鹅：会游泳、会走，但不会飞"""
    def __init__(self, name):
        self.name = name
    
    def slide(self):
        return "I can slide on ice!"

# 使用多重继承
duck = Duck("Donald")
penguin = Penguin("Pingu")

print(f"{duck.name} says: {duck.quack()}")
print(duck.fly())    # 继承自Flyable
print(duck.swim())   # 继承自Swimmable
print(duck.walk())   # 继承自Walkable

print(f"{penguin.name}:")
print(penguin.swim())  # 继承自Swimmable
print(penguin.walk())  # 继承自Walkable
print(penguin.slide()) # Penguin特有的方法
# print(penguin.fly()) # 错误：企鹅不会飞

方法解析顺序（MRO）

在多重继承中，Python使用方法解析顺序（Method Resolution Order, MRO）来确定调用哪个方法： 

class A:
    def method(self):
        return "A.method"

class B(A):
    def method(self):
        return "B.method"

class C(A):
    def method(self):
        return "C.method"

class D(B, C):
    pass

# 查看MRO
print(D.__mro__)
# 输出: (, , , , )

d = D()
print(d.method())  # 输出: B.method，按照MRO顺序查找

多态（Polymorphism）

多态是指不同类的对象对同一消息做出不同的响应。在Python中，多态是通过方法重写和动态绑定实现的。 

class Shape:
    """图形基类"""
    def __init__(self, name):
        self.name = name
    
    def area(self):
        raise NotImplementedError("子类必须实现area方法")
    
    def perimeter(self):
        raise NotImplementedError("子类必须实现perimeter方法")
    
    def describe(self):
        return f"This is a {self.name}"

class Rectangle(Shape):
    """矩形"""
    def __init__(self, width, height):
        super().__init__("Rectangle")
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

class Circle(Shape):
    """圆形"""
    def __init__(self, radius):
        super().__init__("Circle")
        self.radius = radius
    
    def area(self):
        import math
        return math.pi * self.radius ** 2
    
    def perimeter(self):
        import math
        return 2 * math.pi * self.radius

class Triangle(Shape):
    """三角形"""
    def __init__(self, a, b, c):
        super().__init__("Triangle")
        self.a = a
        self.b = b
        self.c = c
    
    def area(self):
        # 使用海伦公式
        import math
        s = (self.a + self.b + self.c) / 2
        return math.sqrt(s * (s - self.a) * (s - self.b) * (s - self.c))
    
    def perimeter(self):
        return self.a + self.b + self.c

# 多态示例 - 同一接口，不同实现
def print_shape_info(shape):
    """打印图形信息的函数，体现多态"""
    print(shape.describe())
    print(f"Area: {shape.area():.2f}")
    print(f"Perimeter: {shape.perimeter():.2f}")
    print("-" * 30)

# 创建不同形状的对象
shapes = [
    Rectangle(5, 3),
    Circle(4),
    Triangle(3, 4, 5)
]

# 使用多态处理不同对象
for shape in shapes:
    print_shape_info(shape)

抽象基类

使用abc模块创建抽象基类，定义接口规范： 

from abc import ABC, abstractmethod

class PaymentMethod(ABC):
    """支付方式的抽象基类"""
    
    @abstractmethod
    def process_payment(self, amount):
        """处理支付的抽象方法"""
        pass
    
    @abstractmethod
    def get_payment_details(self):
        """获取支付详情的抽象方法"""
        pass

class CreditCard(PaymentMethod):
    """信用卡支付"""
    def __init__(self, card_number, card_holder):
        self.card_number = card_number
        self.card_holder = card_holder
    
    def process_payment(self, amount):
        return f"Processing ${amount} payment with credit card ending in {self.card_number[-4:]}"
    
    def get_payment_details(self):
        return f"Credit Card: {self.card_holder}, **** **** **** {self.card_number[-4:]}"

class PayPal(PaymentMethod):
    """PayPal支付"""
    def __init__(self, email):
        self.email = email
    
    def process_payment(self, amount):
        return f"Processing ${amount} payment with PayPal account {self.email}"
    
    def get_payment_details(self):
        return f"PayPal: {self.email}"

class BankTransfer(PaymentMethod):
    """银行转账"""
    def __init__(self, account_number, account_holder):
        self.account_number = account_number
        self.account_holder = account_holder
    
    def process_payment(self, amount):
        return f"Processing ${amount} payment via bank transfer to account {self.account_number}"
    
    def get_payment_details(self):
        return f"Bank Transfer: {self.account_holder}, Account: {self.account_number}"

# 使用多态处理不同类型的支付
def make_payment(payment_method, amount):
    """使用多态处理支付"""
    print(payment_method.get_payment_details())
    print(payment_method.process_payment(amount))
    print("Payment completed successfully!")
    print("-" * 40)

# 创建不同的支付方式
payments = [
    CreditCard("1234567890123456", "John Doe"),
    PayPal("john.doe@example.com"),
    BankTransfer("987654321", "John Doe")
]

# 使用多态处理支付
for payment in payments:
    make_payment(payment, 100.00)

super()函数的高级用法

super()函数用于调用父类的方法，在多重继承中特别有用： 

class Base:
    def __init__(self):
        print("Base.__init__")
    
    def method(self):
        print("Base.method")

class A(Base):
    def __init__(self):
        super().__init__()
        print("A.__init__")
    
    def method(self):
        super().method()
        print("A.method")

class B(Base):
    def __init__(self):
        super().__init__()
        print("B.__init__")
    
    def method(self):
        super().method()
        print("B.method")

class C(A, B):
    def __init__(self):
        super().__init__()
        print("C.__init__")
    
    def method(self):
        super().method()
        print("C.method")

# 创建对象
c = C()
print("\nCalling method:")
c.method()

# 查看MRO
print("\nMRO:", C.__mro__)
