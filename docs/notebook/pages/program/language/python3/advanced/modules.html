模块与包管理 (Modules and Packages)

模块和包是Python组织代码的重要机制。模块是一个包含Python代码的文件，而包是一个包含多个模块的目录。通过模块和包，我们可以更好地组织大型项目，实现代码重用和命名空间管理。

数学工具模块
提供常用的数学计算函数

def add(a, b):
    """加法函数"""
    return a + b

def multiply(a, b):
    """乘法函数"""
    return a * b

def circle_area(radius):
    """计算圆的面积"""
    return PI * radius ** 2

def factorial(n):
    """计算阶乘"""
    if n <= 1:
        return 1
    return n * factorial(n-1)

class LazyImporter:
    """延迟导入器"""
    def __init__(self, module_name):
        self.module_name = module_name
        self._module = None
    
    def __getattr__(self, name):
        if self._module is None:
            self._module = importlib.import_module(self.module_name)
        return getattr(self._module, name)

class PluginManager:
    """插件管理器"""
    def __init__(self):
        self.plugins = {}
    
    def register_plugin(self, name, plugin_func):
        """注册插件"""
        self.plugins[name] = plugin_func
    
    def load_plugins_from_directory(self, directory):
        """从目录加载插件"""
        import glob
        plugin_files = glob.glob(f"{directory}/*.py")
        
        for plugin_file in plugin_files:
            module_name = os.path.splitext(os.path.basename(plugin_file))[0]
            spec = importlib.util.spec_from_file_location(module_name, plugin_file)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            # 假设插件模块有register函数
            if hasattr(module, 'register'):
                module.register(self)

# 4. 子包发现
def discover_subpackages(package_path):
    """发现子包"""
    import pkgutil
    import importlib
    
    subpackages = []
    for importer, modname, ispkg in pkgutil.iter_modules([package_path]):
        if ispkg:
            subpackages.append(modname)
    
    return subpackages

        return package
    except ImportError as e:

        return None

# 6. 包依赖管理
def check_package_dependencies(package_name):
    """检查包依赖"""
    try:
        package = importlib.import_module(package_name)
        if hasattr(package, '__requires__'):
            dependencies = package.__requires__
            print(f"包 {package_name} 的依赖: {dependencies}")
            return dependencies
        else:
            print(f"包 {package_name} 没有声明依赖")
            return []
    except ImportError:
        print(f"包 {package_name} 未安装")
        return None

import sys
import os

def check_virtual_environment():
    """检查是否在虚拟环境中"""
    # 检查VIRTUAL_ENV环境变量
    virtual_env = os.environ.get('VIRTUAL_ENV')
    if virtual_env:
        print(f"当前虚拟环境: {virtual_env}")
        return True
    
    # 检查sys.prefix和sys.base_prefix
    if hasattr(sys, 'base_prefix') and sys.prefix != sys.base_prefix:
        print("在虚拟环境中")
        return True
    
    print("不在虚拟环境中")
    return False

def check_package_version(package_name):
    """检查包版本"""
    try:
        import pkg_resources
        version = pkg_resources.get_distribution(package_name).version
        return version
    except pkg_resources.DistributionNotFound:
        return None
    except ImportError:
        return "无法检查版本"

def parse_requirements(requirements_file):
    """解析requirements.txt文件"""
    dependencies = []
    try:
        with open(requirements_file, 'r') as f:
            for line in f:
                line = line.strip()
                if line and not line.startswith('#'):
                    dependencies.append(line)
    except FileNotFoundError:
        print(f"文件 {requirements_file} 不存在")
    
    return dependencies

# 1. 创建可配置的模块系统
class ConfigurableModule:
    """可配置模块"""
    def __init__(self, config=None):
        self.config = config or {}
        self._plugins = []
    
    def register_plugin(self, plugin):
        """注册插件"""
        self._plugins.append(plugin)
    
    def process(self, data):
        """处理数据"""
        result = data
        for plugin in self._plugins:
            result = plugin.process(result, self.config)
        return result

# 2. 插件架构实现
class PluginManager:
    """插件管理器"""
    def __init__(self, plugin_directory):
        self.plugin_directory = plugin_directory
        self.plugins = {}
        self.load_plugins()
    
    def load_plugins(self):
        """加载插件"""
        import glob
        import importlib.util
        
        plugin_files = glob.glob(f"{self.plugin_directory}/*.py")
        
        for plugin_file in plugin_files:
            module_name = os.path.splitext(os.path.basename(plugin_file))[0]
            if module_name.startswith('_'):
                continue  # 跳过私有模块
                
            spec = importlib.util.spec_from_file_location(module_name, plugin_file)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            # 查找插件类
            for attr_name in dir(module):
                attr = getattr(module, attr_name)
                if (isinstance(attr, type) and 
                    hasattr(attr, 'name') and 
                    hasattr(attr, 'process')):
                    plugin_instance = attr()
                    self.plugins[attr.name] = plugin_instance
    
    def get_plugin(self, name):
        """获取插件"""
        return self.plugins.get(name)
    
    def list_plugins(self):
        """列出所有插件"""
        return list(self.plugins.keys())

# 3. 动态模块加载系统
class DynamicModuleLoader:
    """动态模块加载器"""
    def __init__(self):
        self.loaded_modules = {}
    
    def load_module(self, module_path, module_name=None):
        """动态加载模块"""
        if module_name is None:
            module_name = os.path.splitext(os.path.basename(module_path))[0]
        
        if module_name in self.loaded_modules:
            return self.loaded_modules[module_name]
        
        spec = importlib.util.spec_from_file_location(module_name, module_path)
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        
        self.loaded_modules[module_name] = module
        return module
    
    def unload_module(self, module_name):
        """卸载模块"""
        if module_name in self.loaded_modules:
            del self.loaded_modules[module_name]
    
    def reload_module(self, module_name):
        """重新加载模块"""
        if module_name in self.loaded_modules:
            module = self.loaded_modules[module_name]
            importlib.reload(module)
            return module
        return None

# 4. 包版本管理
class PackageVersionManager:
    """包版本管理器"""
    def __init__(self):
        self.versions = {}
    
    def register_version(self, package_name, version):
        """注册包版本"""
        self.versions[package_name] = version
    
    def check_compatibility(self, package_name, required_version):
        """检查版本兼容性"""
        if package_name not in self.versions:
            return False, "包未注册"
        
        current_version = self.versions[package_name]
        # 简单的版本比较（实际应用中需要更复杂的比较逻辑）
        if current_version == required_version:
            return True, "版本匹配"
        else:
            return False, f"版本不匹配: 当前 {current_version}, 需要 {required_version}"
    
    def get_dependency_tree(self):
        """获取依赖树"""
        import pkg_resources
        
        dependency_tree = {}
        for dist in pkg_resources.working_set:
            dependency_tree[dist.project_name] = {
                'version': dist.version,
                'dependencies': [str(req) for req in dist.requires()]
            }
        
        return dependency_tree

# 4. 版本管理
def version_management_example():
    """版本管理示例"""
    # 在__init__.py中定义版本
    __version__ = "1.2.3"
    
    # 使用语义化版本控制
    # MAJOR.MINOR.PATCH
    # MAJOR: 不兼容的API变更
    # MINOR: 向后兼容的功能添加
    # PATCH: 向后兼容的bug修复
    
    # 版本比较函数
    def compare_versions(v1, v2):
        """比较版本号"""
        def normalize_version(v):
            return [int(x) for x in v.split('.')]
        
        parts1 = normalize_version(v1)
        parts2 = normalize_version(v2)
        
        for p1, p2 in zip(parts1, parts2):
            if p1 < p2:
                return -1
            elif p1 > p2:
                return 1
        
        return 0
    
    print(f"版本比较 1.2.3 vs 1.2.4: {compare_versions('1.2.3', '1.2.4')}")
    print(f"版本比较 2.0.0 vs 1.9.9: {compare_versions('2.0.0', '1.9.9')}")

        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        handler.setFormatter(formatter)
        logger.addHandler(handler)
    
    return logger
