二进制文件处理

二进制文件处理用于处理非文本格式的文件，如图片、音频、视频、可执行文件等。 

# 写入二进制文件
data = bytes([120, 3, 255, 0, 100])
with open('binary.dat', 'wb') as file:
    file.write(data)

# 读取二进制文件
with open('binary.dat', 'rb') as file:
    binary_content = file.read()
    print(f"二进制内容: {binary_content}")
    print(f"十六进制: {binary_content.hex()}")

# 使用BytesIO进行内存操作
buffer = io.BytesIO()
buffer.write(b"Hello, BytesIO!")
buffer.seek(0)
content = buffer.read()
print(f"内存缓冲区内容: {content}")
