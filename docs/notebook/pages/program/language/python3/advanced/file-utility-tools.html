# 计算文件哈希值
def calculate_file_hash(filename, algorithm='sha256'):
    """计算文件的哈希值"""
    hash_obj = hashlib.new(algorithm)
    with open(filename, 'rb') as f:
        for chunk in iter(lambda: f.read(4096), b""):
            hash_obj.update(chunk)
    return hash_obj.hexdigest()

# 创建测试文件
with open('test_hash.txt', 'w') as f:
    f.write('测试哈希值的文件内容')

# 计算哈希值
file_hash = calculate_file_hash('test_hash.txt')
print(f"文件哈希值: {file_hash}")

# 创建压缩文件
files_to_compress = ['test_hash.txt', 'config.ini']

# 创建ZIP文件
with zipfile.ZipFile('archive.zip', 'w') as zipf:
    for file in files_to_compress:
        if os.path.exists(file):
            zipf.write(file)

# 创建TAR文件
with tarfile.open('archive.tar.gz', 'w:gz') as tarf:
    for file in files_to_compress:
        if os.path.exists(file):
            tarf.add(file)

# 解压文件
with zipfile.ZipFile('archive.zip', 'r') as zipf:
    zipf.extractall('extracted_zip')

# 清理测试文件
for filename in ['test_hash.txt', 'config.ini', 'archive.zip', 'archive.tar.gz']:
    if os.path.exists(filename):
        os.remove(filename)
if os.path.exists('extracted_zip'):
    shutil.rmtree('extracted_zip')
