迭代器协议 (Iterator Protocol)

迭代器协议是Python中实现可迭代对象的核心机制。理解迭代器协议对于掌握Python的迭代机制、创建自定义可迭代对象以及高效处理数据序列至关重要。

迭代器协议基础 

# 迭代器协议包含两个方法：
# 1. __iter__() - 返回迭代器对象本身
# 2. __next__() - 返回下一个值，如果没有更多值则抛出StopIteration异常

# 简单的自定义迭代器
class CountDown:
    def __init__(self, start):
        self.start = start
        self.current = start
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.current <= 0:
            raise StopIteration
        value = self.current
        self.current -= 1
        return value

# 使用自定义迭代器
countdown = CountDown(5)
print("倒计时:")
for num in countdown:
    print(num)

# 手动使用迭代器
countdown2 = CountDown(3)
iterator = iter(countdown2)
print(f"手动迭代: {next(iterator)}")  # 3
print(f"手动迭代: {next(iterator)}")  # 2
print(f"手动迭代: {next(iterator)}")  # 1
# print(f"手动迭代: {next(iterator)}")  # StopIteration

# 可迭代对象 vs 迭代器
class NumberRange:
    """可迭代对象（非迭代器）"""
    def __init__(self, start, end):
        self.start = start
        self.end = end
    
    def __iter__(self):
        return NumberRangeIterator(self.start, self.end)

class NumberRangeIterator:
    """迭代器"""
    def __init__(self, start, end):
        self.current = start
        self.end = end
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.current >= self.end:
            raise StopIteration
        value = self.current
        self.current += 1
        return value

    print(num)

    print(num)

# 方法1：实现__iter__方法
class FibonacciIterable:
    """斐波那契数列可迭代对象"""
    def __init__(self, max_count):
        self.max_count = max_count
    
    def __iter__(self):
        return FibonacciIterator(self.max_count)

class FibonacciIterator:
    def __init__(self, max_count):
        self.max_count = max_count
        self.count = 0
        self.a, self.b = 0, 1
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.count >= self.max_count:
            raise StopIteration
        
        if self.count == 0:
            value = self.a
        elif self.count == 1:
            value = self.b
        else:
            self.a, self.b = self.b, self.a + self.b
            value = self.b
        
        self.count += 1
        return value

# 方法2：同时实现__iter__和__next__（对象本身就是迭代器）
class SimpleIterable:
    """简单的可迭代对象"""
    def __init__(self, data):
        self.data = data
        self.index = 0
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        value = self.data[self.index]
        self.index += 1
        return value

# 方法3：使用生成器函数创建可迭代对象
class GeneratorBasedIterable:
    """基于生成器的可迭代对象"""
    def __init__(self, data):
        self.data = data
    
    def __iter__(self):
        for item in self.data:
            yield item * 2

import itertools

# 使用itertools创建强大的迭代器
def itertools_examples():
    """itertools模块示例"""
    
    # 1. 无限迭代器
    print("1. 无限迭代器:")
    
    # count - 无限计数
    counter = itertools.count(10, 2)
    print(f"前5个计数: {[next(counter) for _ in range(5)]}")  # [10, 12, 14, 16, 18]
    
    # cycle - 无限循环
    cycler = itertools.cycle(['A', 'B', 'C'])
    print(f"前8个循环: {[next(cycler) for _ in range(8)]}")  # ['A', 'B', 'C', 'A', 'B', 'C', 'A', 'B']
    
    # repeat - 重复
    repeater = itertools.repeat('Hello', 3)
    print(f"重复3次: {list(repeater)}")  # ['Hello', 'Hello', 'Hello']
    
    # 2. 有限迭代器
    print("\n2. 有限迭代器:")
    
    # accumulate - 累积
    data = [1, 2, 3, 4, 5]
    accumulated = list(itertools.accumulate(data))
    print(f"累积: {accumulated}")  # [1, 3, 6, 10, 15]
    
    # chain - 连接多个可迭代对象
    list1 = [1, 2, 3]
    list2 = [4, 5, 6]
    chained = list(itertools.chain(list1, list2))
    print(f"链式连接: {chained}")  # [1, 2, 3, 4, 5, 6]
    
    # compress - 根据选择器过滤
    data = ['A', 'B', 'C', 'D', 'E']
    selectors = [1, 0, 1, 0, 1]
    compressed = list(itertools.compress(data, selectors))
    print(f"压缩过滤: {compressed}")  # ['A', 'C', 'E']
    
    # dropwhile - 丢弃满足条件的元素
    data = [1, 3, 5, 8, 2, 4, 6]
    dropped = list(itertools.dropwhile(lambda x: x < 5, data))
    print(f"丢弃小于5的元素: {dropped}")  # [8, 2, 4, 6]
    
    # takewhile - 取满足条件的元素
    taken = list(itertools.takewhile(lambda x: x < 5, data))
    print(f"取小于5的元素: {taken}")  # [1, 3]
    
    # groupby - 分组
    data = ['AA', 'BB', 'CC', 'A', 'B', 'C']
    grouped = [(key, list(group)) for key, group in itertools.groupby(data, key=len)]
    print(f"按长度分组: {grouped}")  # [(2, ['AA', 'BB', 'CC']), (1, ['A', 'B', 'C'])]
    
    # 3. 组合迭代器
    print("\n3. 组合迭代器:")
    
    # product - 笛卡尔积
    colors = ['red', 'blue']
    sizes = ['S', 'M', 'L']
    products = list(itertools.product(colors, sizes))
    print(f"颜色和尺寸的组合: {products}")  # [('red', 'S'), ('red', 'M'), ('red', 'L'), ('blue', 'S'), ('blue', 'M'), ('blue', 'L')]
    
    # permutations - 排列
    items = ['A', 'B', 'C']
    perms = list(itertools.permutations(items, 2))
    print(f"A,B,C的2排列: {perms}")  # [('A', 'B'), ('A', 'C'), ('B', 'A'), ('B', 'C'), ('C', 'A'), ('C', 'B')]
    
    # combinations - 组合
    combos = list(itertools.combinations(items, 2))
    print(f"A,B,C的2组合: {combos}")  # [('A', 'B'), ('A', 'C'), ('B', 'C')]
    
    # combinations_with_replacement - 有放回组合
    combos_wr = list(itertools.combinations_with_replacement(items, 2))
    print(f"A,B,C的2有放回组合: {combos_wr}")  # [('A', 'A'), ('A', 'B'), ('A', 'C'), ('B', 'B'), ('B', 'C'), ('C', 'C')]

# 自定义迭代器工具
def take(n, iterable):
    """取前n个元素"""
    return list(itertools.islice(iterable, n))

def enumerate_with_step(iterable, start=0, step=1):
    """带步长的枚举"""
    for i, item in enumerate(iterable):
        yield start + i * step, item

def chunked(iterable, size):
    """分块处理"""
    iterator = iter(iterable)
    while True:
        chunk = list(itertools.islice(iterator, size))
        if not chunk:
            break
        yield chunk

# 使用自定义工具
print("\n自定义迭代器工具:")
numbers = range(10)
print(f"取前5个: {take(5, numbers)}")  # [0, 1, 2, 3, 4]

print("带步长枚举:")
for index, item in enumerate_with_step(data, start=0, step=2):
    print(f"  {index}: {item}")

large_data = range(13)
print("分块处理 (每块3个):")
for chunk in chunked(large_data, 3):
    print(f"  {chunk}")

# 实现反向迭代
class ReversibleRange:
    """可反向迭代的范围"""
    def __init__(self, start, end):
        self.start = start
        self.end = end
    
    def __iter__(self):
        return RangeIterator(self.start, self.end)
    
    def __reversed__(self):
        return RangeIterator(self.end - 1, self.start - 1, -1)

class RangeIterator:
    def __init__(self, start, end, step=1):
        self.current = start
        self.end = end
        self.step = step
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if (self.step > 0 and self.current >= self.end) or \
           (self.step < 0 and self.current <= self.end):
            raise StopIteration
        value = self.current
        self.current += self.step
        return value

# 使用可反向迭代的对象
rev_range = ReversibleRange(1, 6)
print("正向迭代:")
for num in rev_range:
    print(num, end=' ')  # 1 2 3 4 5
print()

print("反向迭代:")
for num in reversed(rev_range):
    print(num, end=' ')  # 5 4 3 2 1
print()

# 自定义迭代顺序
class TreeNode:
    """树节点"""
    def __init__(self, value, left=None, right=None):
        self.value = value
        self.left = left
        self.right = right

class BinaryTree:
    """二叉树"""
    def __init__(self, root):
        self.root = root
    
    def __iter__(self):
        return self.inorder_traversal(self.root)
    
    def inorder_traversal(self, node):
        """中序遍历"""
        if node is not None:
            yield from self.inorder_traversal(node.left)
            yield node.value
            yield from self.inorder_traversal(node.right)
    
    def preorder_traversal(self, node):
        """前序遍历"""
        if node is not None:
            yield node.value
            yield from self.preorder_traversal(node.left)
            yield from self.preorder_traversal(node.right)
    
    def postorder_traversal(self, node):
        """后序遍历"""
        if node is not None:
            yield from self.postorder_traversal(node.left)
            yield from self.postorder_traversal(node.right)
            yield node.value

# 创建二叉树
#       1
#      / \
#     2   3
#    / \
#   4   5
root = TreeNode(1)
root.left = TreeNode(2)
root.right = TreeNode(3)
root.left.left = TreeNode(4)
root.left.right = TreeNode(5)

tree = BinaryTree(root)

print("中序遍历 (默认):")
for value in tree:
    print(value, end=' ')  # 4 2 5 1 3
print()

print("前序遍历:")
for value in tree.preorder_traversal(tree.root):
    print(value, end=' ')  # 1 2 4 5 3
print()

print("后序遍历:")
for value in tree.postorder_traversal(tree.root):
    print(value, end=' ')  # 4 5 2 3 1
print()

# 1. 数据库查询结果迭代器
class DatabaseQueryResult:
    """数据库查询结果迭代器"""
    def __init__(self, connection, query, batch_size=100):
        self.connection = connection
        self.query = query
        self.batch_size = batch_size
        self.cursor = None
        self.current_batch = []
        self.batch_index = 0
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.batch_index >= len(self.current_batch):
            # 获取下一批数据
            if self.cursor is None:
                self.cursor = self.connection.cursor()
                self.cursor.execute(self.query)
            
            self.current_batch = self.cursor.fetchmany(self.batch_size)
            self.batch_index = 0
            
            if not self.current_batch:
                self.cursor.close()
                raise StopIteration
        
        result = self.current_batch[self.batch_index]
        self.batch_index += 1
        return result

# 2. 文件行迭代器（带行号）
class LineIterator:
    """带行号的文件迭代器"""
    def __init__(self, filename, encoding='utf-8'):
        self.filename = filename
        self.encoding = encoding
        self.file = None
        self.line_number = 0
    
    def __enter__(self):
        self.file = open(self.filename, 'r', encoding=self.encoding)
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.file is None:
            raise StopIteration
        
        line = self.file.readline()
        if not line:
            raise StopIteration
        
        self.line_number += 1
        return self.line_number, line.rstrip('\n')

# 3. 分页迭代器
class PaginationIterator:
    """分页迭代器"""
    def __init__(self, data, page_size=10):
        self.data = list(data)
        self.page_size = page_size
        self.current_page = 0
    
    def __iter__(self):
        return self
    
    def __next__(self):
        start_index = self.current_page * self.page_size
        end_index = start_index + self.page_size
        
        if start_index >= len(self.data):
            raise StopIteration
        
        page = self.data[start_index:end_index]
        self.current_page += 1
        return page
    
    def __len__(self):
        return (len(self.data) + self.page_size - 1) // self.page_size
    
    def reset(self):
        """重置迭代器"""
        self.current_page = 0

# 4. 过滤迭代器
class FilterIterator:
    """过滤迭代器"""
    def __init__(self, iterable, predicate):
        self.iterator = iter(iterable)
        self.predicate = predicate
    
    def __iter__(self):
        return self
    
    def __next__(self):
        while True:
            item = next(self.iterator)
            if self.predicate(item):
                return item

# 5. 转换迭代器
class TransformIterator:
    """转换迭代器"""
    def __init__(self, iterable, transform_func):
        self.iterator = iter(iterable)
        self.transform_func = transform_func
    
    def __iter__(self):
        return self
    
    def __next__(self):
        item = next(self.iterator)
        return self.transform_func(item)

# 1. 正确处理StopIteration
class SafeIterator:
    """安全的迭代器"""
    def __init__(self, data):
        self.data = data
        self.index = 0
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        value = self.data[self.index]
        self.index += 1
        return value

# 2. 提供清晰的文档
class DocumentedIterator:
    """带文档的迭代器示例"""
    def __init__(self, start, end):
        """
        创建一个范围迭代器
        
        Args:
            start: 起始值（包含）
            end: 结束值（不包含）
        """
        self.start = start
        self.end = end
        self.current = start
    
    def __iter__(self):
        """返回迭代器对象本身"""
        return self
    
    def __next__(self):
        """返回下一个值
        
        Returns:
            当前值
            
        Raises:
            StopIteration: 当没有更多值时
        """
        if self.current >= self.end:
            raise StopIteration
        value = self.current
        self.current += 1
        return value

# 3. 支持多次迭代
class ReusableIterable:
    """可重复使用的可迭代对象"""
    def __init__(self, data):
        self.data = data
    
    def __iter__(self):
        # 每次调用__iter__都返回新的迭代器
        return SimpleIterator(self.data)

class SimpleIterator:
    def __init__(self, data):
        self.data = data
        self.index = 0
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        value = self.data[self.index]
        self.index += 1
        return value

# 4. 性能优化
class EfficientIterator:
    """高效迭代器"""
    def __init__(self, data):
        # 将数据转换为列表以确保可以多次迭代
        self.data = list(data) if not isinstance(data, (list, tuple)) else data
    
    def __iter__(self):
        # 使用生成器函数，避免创建额外的迭代器类
        for item in self.data:
            yield item * 2

# 5. 错误处理
class RobustIterator:
    """健壮的迭代器"""
    def __init__(self, data):
        if data is None:
            raise ValueError("数据不能为None")
        self.data = data
        self.index = 0
    
    def __iter__(self):
        return self
    
    def __next__(self):
        try:
            if self.index >= len(self.data):
                raise StopIteration
            value = self.data[self.index]
            self.index += 1
            return value
        except TypeError:
            raise StopIteration("数据不支持索引操作")
        except IndexError:
            raise StopIteration("索引超出范围")
