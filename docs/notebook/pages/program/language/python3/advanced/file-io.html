文件操作 (File I/O)

文件操作是Python编程中的重要部分，用于读取、写入和管理文件系统中的数据。Python提供了多种方式来处理文件操作，包括文本文件、二进制文件、JSON、CSV等格式的处理。

基本文件操作 

# 打开和关闭文件
# 传统方式
file = open('example.txt', 'w')
file.write('Hello, World!')
file.close()

# 推荐方式：使用上下文管理器
with open('example.txt', 'w') as file:
    file.write('Hello, World!')

# 文件打开模式
# 'r' - 读取（默认）
# 'w' - 写入（覆盖）
# 'a' - 追加
# 'x' - 独占创建
# 'b' - 二进制模式
# 't' - 文本模式（默认）
# '+' - 读写模式

# 读取文件
with open('example.txt', 'r') as file:
    content = file.read()
    print(content)

# 逐行读取
with open('example.txt', 'r') as file:
    for line in file:
        print(line.strip())

# 读取所有行到列表
with open('example.txt', 'r') as file:
    lines = file.readlines()
    print(lines)

文本文件处理 

# 写入文本文件
with open('story.txt', 'w', encoding='utf-8') as file:
    file.write("这是一个故事\n")
    file.write("从前有座山\n")
    file.write("山里有座庙\n")

# 追加文本
with open('story.txt', 'a', encoding='utf-8') as file:
    file.write("庙里有个老和尚\n")

# 读取文本文件
with open('story.txt', 'r', encoding='utf-8') as file:
    content = file.read()
    print("文件内容:")
    print(content)

# 逐行处理
with open('story.txt', 'r', encoding='utf-8') as file:
    for i, line in enumerate(file, 1):
        print(f"第{i}行: {line.strip()}")

# 使用seek和tell
with open('story.txt', 'r', encoding='utf-8') as file:
    print(f"当前位置: {file.tell()}")
    line = file.readline()
    print(f"读取一行后位置: {file.tell()}")
    file.seek(0)  # 回到文件开头
    print(f"回到开头后位置: {file.tell()}")

CSV文件处理 

import csv

# 写入CSV文件
data = [
    ['姓名', '年龄', '城市'],
    ['张三', 25, '北京'],
    ['李四', 30, '上海'],
    ['王五', 28, '广州']
]

with open('people.csv', 'w', newline='', encoding='utf-8') as file:
    writer = csv.writer(file)
    writer.writerows(data)

# 读取CSV文件
with open('people.csv', 'r', encoding='utf-8') as file:
    reader = csv.reader(file)
    for row in reader:
        print(row)

# 使用DictReader和DictWriter
with open('people.csv', 'r', encoding='utf-8') as file:
    reader = csv.DictReader(file)
    for row in reader:
        print(f"{row['姓名']}，{row['年龄']}岁，来自{row['城市']}")

# 写入字典数据
people = [
    {'姓名': '赵六', '年龄': 35, '城市': '深圳'},
    {'姓名': '钱七', '年龄': 22, '城市': '杭州'}
]

with open('people2.csv', 'w', newline='', encoding='utf-8') as file:
    fieldnames = ['姓名', '年龄', '城市']
    writer = csv.DictWriter(file, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerows(people)

文件和目录操作 

import os
import shutil
import glob
from pathlib import Path

# 获取当前工作目录
current_dir = os.getcwd()
print(f"当前目录: {current_dir}")

# 创建目录
os.makedirs('new_folder/subfolder', exist_ok=True)

# 检查文件或目录是否存在
print(f"文件存在: {os.path.exists('example.txt')}")
print(f"是文件: {os.path.isfile('example.txt')}")
print(f"是目录: {os.path.isdir('new_folder')}")

# 获取文件信息
if os.path.exists('example.txt'):
    print(f"文件大小: {os.path.getsize('example.txt')} 字节")
    print(f"最后修改时间: {os.path.getmtime('example.txt')}")

# 重命名文件
if os.path.exists('example.txt'):
    os.rename('example.txt', 'renamed_example.txt')

# 复制文件
if os.path.exists('renamed_example.txt'):
    shutil.copy('renamed_example.txt', 'copy_example.txt')

# 移动文件
if os.path.exists('copy_example.txt'):
    shutil.move('copy_example.txt', 'new_folder/moved_example.txt')

# 删除文件
if os.path.exists('renamed_example.txt'):
    os.remove('renamed_example.txt')

# 删除目录
if os.path.exists('new_folder'):
    shutil.rmtree('new_folder')

# 使用pathlib（更现代的方式）
path = Path('test_directory')
path.mkdir(exist_ok=True)

file_path = path / 'test_file.txt'
file_path.write_text('使用pathlib创建的文件', encoding='utf-8')

print(f"文件内容: {file_path.read_text(encoding='utf-8')}")
print(f"文件父目录: {file_path.parent}")
print(f"文件后缀: {file_path.suffix}")
print(f"文件stem: {file_path.stem}")

# 清理
if file_path.exists():
    file_path.unlink()
if path.exists():
    path.rmdir()

大文件处理 

# 创建大文件用于测试
def create_large_file(filename, size_mb=10):
    """创建指定大小的测试文件"""
    chunk_size = 1024 * 1024  # 1MB
    chunk = b'x' * chunk_size
    
    with open(filename, 'wb') as f:
        for _ in range(size_mb):
            f.write(chunk)
    print(f"创建了 {size_mb}MB 的文件: {filename}")

# 逐行读取大文件
def read_large_file_line_by_line(filename):
    """逐行读取大文件"""
    line_count = 0
    with open(filename, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            line_count += 1
            if line_count % 100000 == 0:
                print(f"已读取 {line_count} 行")
    return line_count

# 分块读取大文件
def read_large_file_in_chunks(filename, chunk_size=8192):
    """分块读取大文件"""
    total_bytes = 0
    with open(filename, 'rb') as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            total_bytes += len(chunk)
            # 处理chunk数据
    return total_bytes

# 使用生成器处理大文件
def process_large_file(filename):
    """使用生成器逐行处理大文件"""
    def line_generator(filename):
        with open(filename, 'r', encoding='utf-8', errors='ignore') as f:
            for line in f:
                yield line.strip()
    
    # 处理每一行
    for i, line in enumerate(line_generator(filename)):
        if i % 10000 == 0:
            print(f"处理第 {i} 行: {line[:50]}...")

# 内存映射文件（适用于超大文件）
import mmap

def process_with_mmap(filename):
    """使用内存映射处理大文件"""
    with open(filename, 'r+b') as f:
        # 内存映射整个文件
        with mmap.mmap(f.fileno(), 0) as mm:
            # 可以像处理字符串一样处理内存映射
            lines = 0
            while True:
                line = mm.readline()
                if not line:
                    break
                lines += 1
                if lines % 100000 == 0:
                    print(f"内存映射方式处理第 {lines} 行")
            return lines

# 计算文件哈希值
def calculate_file_hash(filename, algorithm='sha256'):
    """计算文件的哈希值"""
    hash_obj = hashlib.new(algorithm)
    with open(filename, 'rb') as f:
        for chunk in iter(lambda: f.read(4096), b""):
            hash_obj.update(chunk)
    return hash_obj.hexdigest()

# 创建测试文件

    f.write('测试哈希值的文件内容')

# 计算哈希值

print(f"文件哈希值: {file_hash}")
