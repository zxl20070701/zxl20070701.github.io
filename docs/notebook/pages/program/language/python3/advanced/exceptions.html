异常处理 (Exception Handling)

异常处理是Python中处理程序运行时错误的重要机制。通过异常处理，我们可以优雅地处理错误情况，防止程序崩溃，并提供有意义的错误信息。

基本异常处理 

# 基本try-except结构

    print("不能除以零!")

自定义异常 

# 自定义异常类
class CustomError(Exception):
    """自定义异常基类"""
    pass

class ValidationError(CustomError):
    """验证错误"""
    def __init__(self, message, field=None):
        super().__init__(message)
        self.field = field

class InsufficientFundsError(CustomError):
    """余额不足错误"""
    def __init__(self, balance, amount):
        self.balance = balance
        self.amount = amount
        super().__init__(f"余额不足: 当前余额 {balance}, 需要 {amount}")

# 使用自定义异常
class BankAccount:
    def __init__(self, initial_balance=0):
        self.balance = initial_balance
    
    def deposit(self, amount):
        if amount <= 0:
            raise ValidationError("存款金额必须大于零", "amount")
        self.balance += amount
        return self.balance
    
    def withdraw(self, amount):
        if amount <= 0:
            raise ValidationError("取款金额必须大于零", "amount")
        if amount > self.balance:
            raise InsufficientFundsError(self.balance, amount)
        self.balance -= amount
        return self.balance

# 测试银行账户
account = BankAccount(100)

try:
    account.deposit(50)
    print(f"存款后余额: {account.balance}")
    
    account.withdraw(200)
except ValidationError as e:
    print(f"验证错误: {e}")
    if e.field:
        print(f"错误字段: {e.field}")
except InsufficientFundsError as e:
    print(f"余额不足错误: {e}")
    print(f"当前余额: {e.balance}, 请求金额: {e.amount}")

异常链和重新抛出 

# 异常链
def divide_numbers(a, b):
    try:
        return a / b
    except ZeroDivisionError as e:
        # 创建新的异常并保留原始异常
        raise ValueError("参数b不能为零") from e

try:
    result = divide_numbers(10, 0)
except ValueError as e:
    print(f"捕获到值错误: {e}")
    print(f"原始异常: {e.__cause__}")

# 重新抛出异常
def process_data(data):
    try:
        # 一些处理逻辑
        if not data:
            raise ValueError("数据不能为空")
        return data.upper()
    except ValueError:
        print("处理数据时发生值错误")
        raise  # 重新抛出相同的异常

try:
    process_data("")
except ValueError as e:
    print(f"最终捕获: {e}")

# 抑制异常
def may_fail():
    try:
        # 一些可能失败的操作
        return 10 / 0
    except ZeroDivisionError:
        print("忽略除零错误")
        return None

result = may_fail()
print(f"结果: {result}")

    try:
        return a / b
    except ZeroDivisionError:  # 具体异常

# 2. 使用finally清理资源
def read_file_safely(filename):
    file = None
    try:
        file = open(filename, 'r')
        return file.read()
    except FileNotFoundError:
        print(f"文件 {filename} 未找到")
        return None
    except IOError:
        print(f"读取文件 {filename} 时发生IO错误")
        return None
    finally:
        if file:
            file.close()
            print("文件已关闭")

# 1. EAFP (Easier to Ask for Forgiveness than Permission)
def read_file_eafp(filename):
    """EAFP模式：先尝试，失败再处理"""
    try:
        with open(filename, 'r') as file:
            return file.read()
    except FileNotFoundError:
        return "文件不存在"
    except IOError:
        return "读取文件失败"

# 2. LBYL (Look Before You Leap)
def read_file_lbyl(filename):
    """LBYL模式：先检查，再执行"""
    import os
    if not os.path.exists(filename):
        return "文件不存在"
    if not os.access(filename, os.R_OK):
        return "文件不可读"
    
    try:
        with open(filename, 'r') as file:
            return file.read()
    except IOError:
        return "读取文件失败"

# 3. 重试模式
import time
import random

def retry_operation(operation, max_attempts=3, delay=1):
    """重试操作"""
    for attempt in range(max_attempts):
        try:
            return operation()
        except Exception as e:
            if attempt == max_attempts - 1:
                raise
            print(f"尝试 {attempt + 1} 失败: {e}，{delay}秒后重试...")
            time.sleep(delay)

# 测试重试
def unreliable_operation():
    """模拟不可靠操作"""
    if random.random() < 0.7:  # 70%概率失败
        raise ConnectionError("网络连接失败")
    return "操作成功"

try:
    result = retry_operation(unreliable_operation, max_attempts=3, delay=0.5)
    print(result)
except ConnectionError as e:
    print(f"所有重试都失败: {e}")

# 4. 回退模式
def get_data_with_fallback(primary_source, fallback_source):
    """带回退的数据获取"""
    try:
        return primary_source()
    except Exception as e:
        print(f"主数据源失败: {e}")
        try:
            return fallback_source()
        except Exception as e2:
            print(f"备用数据源也失败: {e2}")
            raise

# 5. 批量处理异常
def process_items(items):
    """批量处理项目，收集所有异常"""
    results = []
    exceptions = []
    
    for item in items:
        try:
            # 模拟处理逻辑
            if item < 0:
                raise ValueError(f"项目 {item} 不能为负数")
            result = item * 2
            results.append(result)
        except Exception as e:
            exceptions.append((item, e))
    
    return results, exceptions

# 测试批量处理
items = [1, -2, 3, -4, 5]
results, errors = process_items(items)

print(f"成功处理的结果: {results}")
print(f"处理错误:")
for item, error in errors:
    print(f"  项目 {item}: {error}")

# 1. 使用traceback模块获取详细信息
import traceback

def risky_function():
    try:
        # 多层嵌套调用
        def level3():
            raise ValueError("这是第三层的错误")
        
        def level2():
            level3()
        
        def level1():
            level2()
        
        level1()
    except Exception as e:
        print("异常类型:", type(e).__name__)
        print("异常消息:", str(e))
        print("\n完整回溯:")
        print(traceback.format_exc())
        
        # 只获取最后一行
        print("\n最后调用:")
        print(traceback.format_exc(limit=1))

# 2. 使用sys.exc_info()获取异常信息
import sys

def handle_exception():
    try:
        raise TypeError("类型错误示例")
    except:
        exc_type, exc_value, exc_traceback = sys.exc_info()
        print(f"异常类型: {exc_type.__name__}")
        print(f"异常值: {exc_value}")
        print(f"追踪对象: {exc_traceback}")

# 4. 创建异常处理器装饰器
def exception_handler(handler_func=None):
    """异常处理器装饰器"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            try:
                return func(*args, **kwargs)
            except Exception as e:
                if handler_func:
                    return handler_func(e)
                else:
                    print(f"函数 {func.__name__} 发生异常: {e}")
                    return None
        return wrapper
    return decorator

# 使用异常处理器装饰器
@exception_handler()
def divide_safe(a, b):
    return a / b

@exception_handler(lambda e: f"自定义错误处理: {e}")
def process_data(data):
    if not data:
        raise ValueError("数据不能为空")
    return data.upper()

print(divide_safe(10, 2))  # 正常情况
print(divide_safe(10, 0))  # 异常情况

print(process_data("hello"))  # 正常情况
print(process_data(""))      # 异常情况
