比较运算符

比较运算符用于比较两个值之间的关系，返回布尔值True或False。Python支持链式比较，如a < b < c，这在其他许多编程语言中是不常见的。

运算符 描述 示例 结果
== 等于：比较两个对象是否相等 5 == 5 True
!= 不等于：比较两个对象是否不相等 5 != 3 True
> 大于：返回x是否大于y 5 > 3 True
< 小于：返回x是否小于y 3 < 5 True
>= 大于等于：返回x是否大于等于y 5 >= 5 True
<= 小于等于：返回x是否小于等于y 3 <= 5 True

示例

数字比较 

a = 10
b = 20
c = 10

print(f"a = {a}, b = {b}, c = {c}")
print(f"a == b: {a == b}")    # False
print(f"a != b: {a != b}")    # True
print(f"a > b: {a > b}")      # False
print(f"a < b: {a < b}")      # True
print(f"a >= c: {a >= c}")    # True
print(f"a <= c: {a <= c}")    # True

字符串比较 

# 字典序
str1 = "apple"
str2 = "banana"
str3 = "apple"

print(f"'{str1}' == '{str3}': {str1 == str3}")  # True
print(f"'{str1}' != '{str2}': {str1 != str2}")  # True
print(f"'{str1}' < '{str2}': {str1 < str2}")    # True (按字母顺序)
print(f"'{str1}' > '{str2}': {str1 > str2}")    # False

列表比较 

# 逐元素比较
list1 = [1, 2, 3]
list2 = [1, 2, 4]
list3 = [1, 2, 3]

print(f"{list1} == {list3}: {list1 == list3}")  # True
print(f"{list1} != {list2}: {list1 != list2}")  # True
print(f"{list1} < {list2}: {list1 < list2}")    # True

链式比较 

# Python特有
print(f"1 < 2 < 3: {1 < 2 < 3}")        # True
print(f"1 < 3 > 2: {1 < 3 > 2}")        # True
print(f"5 == 5 == 5: {5 == 5 == 5}")    # True
print(f"a == c == 10: {a == c == 10}")  # True

# 等价于
print(f"(1 < 2) and (2 < 3): {(1 < 2) and (2 < 3)}")  # True

不同类型之间的比较 

# 数字比较
print("数字比较:")
print(f"5 == 5.0: {5 == 5.0}")    # True
print(f"5 == 5+0j: {5 == 5+0j}")  # True

# 字符串和数字不能直接比较（Python 3中）
try:
    result = 5 == "5"
    print(f"5 == '5': {result}")  # False
except Exception as e:
    print(f"错误: {e}")

# None比较
print(f"\nNone比较:")
print(f"None == None: {None == None}")  # True
print(f"None != 0: {None != 0}")        # True
print(f"None is None: {None is None}")  # True (推荐使用is)

# 布尔值比较
print(f"\n布尔值比较:")
print(f"True == 1: {True == 1}")      # True
print(f"False == 0: {False == 0}")    # True
print(f"True is 1: {True is 1}")      # False (值相等但对象不同)

对象比较 vs 身份比较 

# 值比较 (==) vs 身份比较 (is)
a = [1, 2, 3]
b = [1, 2, 3]
c = a

print("值比较 vs 身份比较:")
print(f"a: {a}")
print(f"b: {b}")
print(f"c: {c}")
print(f"a == b: {a == b}")    # True (值相等)
print(f"a is b: {a is b}")    # False (不同对象)
print(f"a is c: {a is c}")    # True (同一对象)

# 字符串驻留（string interning）
s1 = "hello"
s2 = "hello"
s3 = "hello world"
s4 = "hello world"

print(f"\n字符串驻留:")
print(f"s1 == s2: {s1 == s2}")  # True
print(f"s1 is s2: {s1 is s2}")  # True (短字符串会被驻留)
print(f"s3 == s4: {s3 == s4}")  # True
print(f"s3 is s4: {s3 is s4}")  # False (长字符串通常不被驻留)

# 数字驻留
x = 256
y = 256
z = 257
w = 257

print(f"\n数字驻留:")
print(f"x is y: {x is y}")  # True (-5到256的整数会被驻留)
print(f"z is w: {z is w}")  # False (超出范围的整数不被驻留)

自定义对象比较 

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __eq__(self, other):
        """定义等于比较"""
        if isinstance(other, Person):
            return self.name == other.name and self.age == other.age
        return False

    def __lt__(self, other):
        """定义小于比较（用于排序）"""
        if isinstance(other, Person):
            return self.age < other.age
        return NotImplemented

    def __repr__(self):
        return f"Person('{self.name}', {self.age})"

# 创建Person对象
p1 = Person("Alice", 25)
p2 = Person("Bob", 30)
p3 = Person("Alice", 25)

print("自定义对象比较:")
print(f"p1: {p1}")
print(f"p2: {p2}")
print(f"p3: {p3}")
print(f"p1 == p3: {p1 == p3}")  # True
print(f"p1 == p2: {p1 == p2}")  # False
print(f"p1 < p2: {p1 < p2}")    # True
print(f"p1 > p2: {p1 > p2}")    # False (自动推断)

# 排序自定义对象
people = [p2, p1, Person("Charlie", 20)]
sorted_people = sorted(people)
print(f"\n按年龄排序: {sorted_people}")

比较运算符优先级 

# 比较运算符优先级低于算术运算符
result1 = 2 + 3 > 4  # 等价于 (2 + 3) > 4
result2 = 2 + (3 > 4)  # 先比较，再加法

print("运算符优先级:")
print(f"2 + 3 > 4 = {result1}")      # True
print(f"2 + (3 > 4) = {result2}")    # 2 (False转换为0)

# 链式比较的结合性
# a < b < c 等价于 (a < b) and (b < c)
a, b, c = 1, 2, 3
chain1 = a < b < c
chain2 = (a < b) and (b < c)
print(f"\n链式比较:")
print(f"a < b < c: {chain1}")          # True
print(f"(a < b) and (b < c): {chain2}")  # True
print(f"等价: {chain1 == chain2}")     # True

# 复杂的链式比较
x = 5
complex_chain = 1 < x <= 5 != 6
print(f"1 < {x} <= 5 != 6: {complex_chain}")  # True
