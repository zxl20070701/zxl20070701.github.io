算术运算符

算术运算符用于执行基本的数学运算，包括加、减、乘、除等。Python中的算术运算符不仅适用于数字类型，还可以用于字符串、列表等序列类型的特定操作。

运算符 描述 示例 结果
+ 加法：两个对象相加 5 + 3 8
- 减法：得到负数或一个数减去另一个数 10 - 4 6
* 乘法：两个数相乘或字符串重复 3 * 4 12
/ 除法：返回浮点数结果 7 / 2 3.5
// 整除：向下取整的除法 7 // 2 3
% 取模：返回除法的余数 7 % 3 1
** 幂运算：返回x的y次幂 2 ** 3 8

示例

基本算术运算 

a = 10
b = 3

print(f"a = {a}, b = {b}")
print(f"a + b = {a + b}")    # 13
print(f"a - b = {a - b}")    # 7
print(f"a * b = {a * b}")    # 30
print(f"a / b = {a / b}")    # 3.333...
print(f"a // b = {a // b}")  # 3 (整除)
print(f"a % b = {a % b}")    # 1 (取模)
print(f"a ** b = {a ** b}")  # 1000 (10的3次方)

字符串操作 

str1 = "Hello"
str2 = "World"

print(f"str1 + str2 = {str1 + ' ' + str2}")  # Hello World
print(f"str1 * 3 = {str1 * 3}")  # HelloHelloHello

列表操作 

list1 = [1, 2, 3]
list2 = [4, 5, 6]

print(f"list1 + list2 = {list1 + list2}")  # [1, 2, 3, 4, 5, 6]
print(f"list1 * 2 = {list1 * 2}")  # [1, 2, 3, 1, 2, 3]

复合赋值运算符 

x = 5
print(f"初始值: x = {x}")

x += 3   # 等价于 x = x + 3
print(f"x += 3 -> x = {x}")  # 8

x -= 2   # 等价于 x = x - 2
print(f"x -= 2 -> x = {x}")  # 6

x *= 4   # 等价于 x = x * 4
print(f"x *= 4 -> x = {x}")  # 24

x //= 3  # 等价于 x = x // 3
print(f"x //= 3 -> x = {x}")  # 8

x %= 5   # 等价于 x = x % 5
print(f"x %= 5 -> x = {x}")  # 3

x **= 2  # 等价于 x = x ** 2
print(f"x **= 2 -> x = {x}")  # 9

浮点数精度问题 

a = 0.1
b = 0.2
c = a + b

print(f"0.1 + 0.2 = {c}")  # 0.30000000000000004
print(f"0.1 + 0.2 == 0.3: {c == 0.3}")  # False

# 解决方案1：使用round()
print(f"round(0.1 + 0.2, 1) == 0.3: {round(c, 1) == 0.3}")  # True

# 解决方案2：使用decimal模块
from decimal import Decimal
d1 = Decimal('0.1')
d2 = Decimal('0.2')
d3 = d1 + d2
print(f"Decimal: 0.1 + 0.2 = {d3}")  # 0.3
print(f"Decimal: 0.1 + 0.2 == 0.3: {d3 == Decimal('0.3')}")  # True

# 解决方案3：使用math.isclose()
import math
print(f"math.isclose(0.1 + 0.2, 0.3): {math.isclose(0.1 + 0.2, 0.3)}")  # True

数学函数和常量 

import math

# 常用数学函数
print("常用数学函数:")
print(f"abs(-5) = {abs(-5)}")  # 绝对值
print(f"round(3.14159, 2) = {round(3.14159, 2)}")  # 四舍五入
print(f"max(1, 5, 3, 9, 2) = {max(1, 5, 3, 9, 2)}")  # 最大值
print(f"min(1, 5, 3, 9, 2) = {min(1, 5, 3, 9, 2)}")  # 最小值
print(f"sum([1, 2, 3, 4, 5]) = {sum([1, 2, 3, 4, 5])}")  # 求和

# 数学常量
print("\n数学常量:")
print(f"π = {math.pi}")
print(f"e = {math.e}")
print(f"∞ = {math.inf}")
print(f"-∞ = {-math.inf}")
print(f"NaN = {math.nan}")

# 数学函数
print("\n高级数学函数:")
print(f"sqrt(16) = {math.sqrt(16)}")  # 平方根
print(f"pow(2, 3) = {math.pow(2, 3)}")  # 幂运算
print(f"ceil(3.2) = {math.ceil(3.2)}")  # 向上取整
print(f"floor(3.8) = {math.floor(3.8)}")  # 向下取整
print(f"trunc(3.8) = {math.trunc(3.8)}")  # 截断小数部分

运算符优先级 

# 运算符优先级示例
result1 = 2 + 3 * 4  # 先乘后加
result2 = (2 + 3) * 4  # 括号改变优先级

print(f"2 + 3 * 4 = {result1}")  # 14
print(f"(2 + 3) * 4 = {result2}")  # 20

# 复杂表达式
complex_expr = 2 + 3 * 4 ** 2 / 2 - 1
print(f"2 + 3 * 4 ** 2 / 2 - 1 = {complex_expr}")  # 25.0

# 使用括号明确优先级
explicit = 2 + ((3 * (4 ** 2)) / 2) - 1
print(f"显式括号: {explicit}")  # 25.0

# 运算符优先级表（从高到低）
precedence = """
** (指数运算)
*, /, //, % (乘、除、整除、取模)
+, - (加、减)
"""
print(f"\n运算符优先级（从高到低）:\n{precedence}")
