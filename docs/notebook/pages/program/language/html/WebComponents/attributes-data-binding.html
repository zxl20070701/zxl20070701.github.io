属性与数据绑定

组件属性的基本概念

Web Components中的属性（Attributes）是HTML元素的标准特性，用于传递静态数据到组件。属性在HTML中设置，并且可以通过getAttribute()和setAttribute()方法进行读取和修改。

属性主要用于初始配置，而属性（Properties）则用于动态数据交互。

基本的属性使用 

class AttributeComponent extends HTMLElement {
    constructor() {
        super();
        const shadow = this.attachShadow({mode: 'open'});
        
        shadow.innerHTML = `
            
            
            
属性组件

            

                
标题: 未设置

                
颜色: 未设置

                
大小: 未设置

                
是否启用: 未设置

            

        `;
        
        this.updateContent();
    }
    
    // 当组件被插入到DOM时调用
    connectedCallback() {
        this.updateContent();
    }
    
    // 更新组件内容
    updateContent() {
        const title = this.getAttribute('title') || '默认标题';
        const color = this.getAttribute('color') || 'black';
        const size = this.getAttribute('size') || 'medium';
        const enabled = this.hasAttribute('enabled');
        
        const shadow = this.shadowRoot;
        shadow.getElementById('title').textContent = title;
        shadow.getElementById('color').textContent = color;
        shadow.getElementById('size').textContent = size;
        shadow.getElementById('enabled').textContent = enabled ? '是' : '否';
        
        // 应用颜色样式
        shadow.getElementById('color').style.color = color;
    }
    
    // 监听属性变化
    static get observedAttributes() {
        return ['title', 'color', 'size'];
    }
    
    // 当观察的属性变化时调用
    attributeChangedCallback(name, oldValue, newValue) {
        if (oldValue !== newValue) {
            this.updateContent();
        }
    }
}

customElements.define('attribute-component', AttributeComponent);

属性与属性的区别

在Web Components中，属性（Attributes）和属性（Properties）是两个不同的概念：

  • 属性（Attributes） ：HTML元素上的特性，始终是字符串类型
  • 属性（Properties） ：JavaScript对象上的属性，可以是任何类型 
class PropertyComponent extends HTMLElement {
    constructor() {
        super();
        const shadow = this.attachShadow({mode: 'open'});
        
        shadow.innerHTML = `
            
            
            

                
属性与属性对比

                

                    
属性值 (attribute): 未设置

                    
属性值 (property): 未设置

                

                
                

                    
                    
                

                
                

                    
                    
                

            

        `;
        
        this.bindEvents();
    }
    
    bindEvents() {
        const shadow = this.shadowRoot;
        
        shadow.getElementById('set-attr').addEventListener('click', () => {
            const value = shadow.getElementById('attr-input').value;
            this.setAttribute('data-value', value);
        });
        
        shadow.getElementById('set-prop').addEventListener('click', () => {
            const value = shadow.getElementById('prop-input').value;
            this.dataValue = value;
        });
    }
    
    // 观察的属性
    static get observedAttributes() {
        return ['data-value'];
    }
    
    // 属性变化回调
    attributeChangedCallback(name, oldValue, newValue) {
        if (name === 'data-value') {
            this.shadowRoot.getElementById('attr-value').textContent = newValue || '未设置';
        }
    }
    
    // 定义属性
    get dataValue() {
        return this._dataValue;
    }
    
    set dataValue(value) {
        this._dataValue = value;
        this.shadowRoot.getElementById('prop-value').textContent = value || '未设置';
    }
}

customElements.define('property-component', PropertyComponent);

数据绑定

数据绑定是将组件的属性或属性与UI元素关联起来的过程。当数据变化时，UI会自动更新。 

class DataBindingComponent extends HTMLElement {
    constructor() {
        super();
        const shadow = this.attachShadow({mode: 'open'});
        
        shadow.innerHTML = `
            
            
            

                
数据绑定示例

                
                

                    
                    
                

                
                

                    
                    
                

                
                

                    
                    
                

                
                

                    
                    
                

                
                

                    
                    
                

                
                

                    
预览:

                    

                        请填写上面的表单...
                    

                

            

        `;
        
        this.data = {
            name: '',
            email: '',
            age: '',
            gender: '',
            bio: ''
        };
        
        this.bindEvents();
        this.updatePreview();
    }
    
    bindEvents() {
        const shadow = this.shadowRoot;
        
        // 绑定所有输入框的事件
        const inputs = shadow.querySelectorAll('input, select, textarea');
        inputs.forEach(input => {
            input.addEventListener('input', (e) => {
                const key = e.target.id;
                const value = e.target.value;
                this.data[key] = value;
                this.updatePreview();
            });
        });
    }
    
    updatePreview() {
        const shadow = this.shadowRoot;
        const preview = shadow.getElementById('preview-content');
        
        if (Object.values(this.data).every(value => !value)) {
            preview.innerHTML = '请填写上面的表单...';
            return;
        }
        
        let html = '';
        
        if (this.data.name) {
            html += `
姓名: ${this.data.name}
`;
        }
        
        if (this.data.email) {
            html += `
邮箱: ${this.data.email}
`;
        }
        
        if (this.data.age) {
            html += `
年龄: ${this.data.age}岁
`;
        }
        
        if (this.data.gender) {
            const genderMap = {
                'male': '男',
                'female': '女',
                'other': '其他'
            };
            html += `
性别: ${genderMap[this.data.gender]}
`;
        }
        
        if (this.data.bio) {
            html += `
个人简介: ${this.data.bio}
`;
        }
        
        preview.innerHTML = html;
    }
}

customElements.define('data-binding-component', DataBindingComponent);

双向数据绑定

双向数据绑定是指数据变化会更新UI，UI变化也会更新数据。 

class TwoWayBindingComponent extends HTMLElement {
    constructor() {
        super();
        const shadow = this.attachShadow({mode: 'open'});
        
        shadow.innerHTML = `
            
            
            

                

                    
双向数据绑定计数器

                    
0

                

                
                

                    
                    
                    
                

                
                

                    
                    
                

            

        `;
        
        this.count = 0;
        this.bindEvents();
        this.updateUI();
    }
    
    bindEvents() {
        const shadow = this.shadowRoot;
        
        // 按钮事件
        shadow.getElementById('decrease').addEventListener('click', () => {
            this.count--;
            this.updateUI();
        });
        
        shadow.getElementById('increase').addEventListener('click', () => {
            this.count++;
            this.updateUI();
        });
        
        shadow.getElementById('reset').addEventListener('click', () => {
            this.count = 0;
            this.updateUI();
        });
        
        // 输入框事件
        shadow.getElementById('count-input').addEventListener('input', (e) => {
            const value = parseInt(e.target.value) || 0;
            this.count = value;
            this.updateUI();
        });
    }
    
    updateUI() {
        const shadow = this.shadowRoot;
        shadow.getElementById('count').textContent = this.count;
        shadow.getElementById('count-input').value = this.count;
        
        // 触发自定义事件
        this.dispatchEvent(new CustomEvent('count-changed', {
            detail: { count: this.count },
            bubbles: true
        }));
    }
    
    // 提供属性访问器
    get value() {
        return this.count;
    }
    
    set value(newValue) {
        this.count = parseInt(newValue) || 0;
        this.updateUI();
    }
}

customElements.define('two-way-binding-component', TwoWayBindingComponent);

高级属性处理

处理复杂的数据类型和属性验证： 

class AdvancedAttributeComponent extends HTMLElement {
    constructor() {
        super();
        const shadow = this.attachShadow({mode: 'open'});
        
        shadow.innerHTML = `
            
            
            

                
高级属性处理

                
                

                    
配置数据:

                    
未设置

                

                
                

                    
JSON输出:

                    
{}

                

                
                

            

        `;
        
        this._config = {};
        this.updateDisplay();
    }
    
    // 观察的属性
    static get observedAttributes() {
        return ['config', 'theme', 'features'];
    }
    
    // 属性变化回调
    attributeChangedCallback(name, oldValue, newValue) {
        if (oldValue !== newValue) {
            this.updateProperty(name, newValue);
        }
    }
    
    // 更新属性值
    updateProperty(name, value) {
        try {
            switch (name) {
                case 'config':
                    this._config = value ? JSON.parse(value) : {};
                    break;
                case 'theme':
                    this._theme = value || 'default';
                    break;
                case 'features':
                    this._features = value ? value.split(',').map(f => f.trim()) : [];
                    break;
            }
            this.updateDisplay();
            this.clearError();
        } catch (error) {
            this.showError(`解析 ${name} 失败: ${error.message}`);
        }
    }
    
    updateDisplay() {
        const shadow = this.shadowRoot;
        
        // 显示配置
        const configDisplay = shadow.getElementById('config-display');
        configDisplay.innerHTML = `
            
主题: ${this._theme || 'default'}

            
功能: ${(this._features || []).join(', ') || '无'}

            
配置项数: ${Object.keys(this._config).length}

        `;
        
        // 显示JSON
        const jsonOutput = shadow.getElementById('json-output');
        const fullConfig = {
            theme: this._theme || 'default',
            features: this._features || [],
            config: this._config
        };
        jsonOutput.textContent = JSON.stringify(fullConfig, null, 2);
    }
    
    showError(message) {
        const shadow = this.shadowRoot;
        const errorElement = shadow.getElementById('error-message');
        errorElement.innerHTML = `
${message}
`;
    }
    
    clearError() {
        const shadow = this.shadowRoot;
        const errorElement = shadow.getElementById('error-message');
        errorElement.innerHTML = '';
    }
    
    // 属性访问器
    get config() {
        return this._config;
    }
    
    set config(value) {
        this._config = value;
        this.setAttribute('config', JSON.stringify(value));
    }
    
    get theme() {
        return this._theme;
    }
    
    set theme(value) {
        this._theme = value;
        this.setAttribute('theme', value);
    }
    
    get features() {
        return this._features;
    }
    
    set features(value) {
        this._features = Array.isArray(value) ? value : [];
        this.setAttribute('features', this._features.join(','));
    }
}

customElements.define('advanced-attribute-component', AdvancedAttributeComponent);
