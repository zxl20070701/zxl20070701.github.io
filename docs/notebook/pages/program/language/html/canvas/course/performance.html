性能优化

性能优化的重要性

Canvas应用，特别是游戏和数据可视化，对性能要求很高。优化Canvas性能可以显著提升用户体验，特别是在移动设备上。

减少绘制调用

批量绘制 

// 不好的做法：多次调用fillRect
for (let i = 0; i < 1000; i++) {
    ctx.fillStyle = colors[i];
    ctx.fillRect(x[i], y[i], width[i], height[i]);
}

// 好的做法：先按颜色分组，再批量绘制
const colorGroups = {};
for (let i = 0; i < 1000; i++) {
    const color = colors[i];
    if (!colorGroups[color]) {
        colorGroups[color] = [];
    }
    colorGroups[color].push({x: x[i], y: y[i], w: width[i], h: height[i]});
}

for (const color in colorGroups) {
    ctx.fillStyle = color;
    const rects = colorGroups[color];
    for (const rect of rects) {
        ctx.fillRect(rect.x, rect.y, rect.w, rect.h);
    }
}

脏矩形优化 

let dirtyRect = null;

function markDirty(x, y, width, height) {
    if (!dirtyRect) {
        dirtyRect = {x, y, width, height};
    } else {
        // 扩展脏矩形以包含新区域
        const right = Math.max(dirtyRect.x + dirtyRect.width, x + width);
        const bottom = Math.max(dirtyRect.y + dirtyRect.height, y + height);
        
        dirtyRect.x = Math.min(dirtyRect.x, x);
        dirtyRect.y = Math.min(dirtyRect.y, y);
        dirtyRect.width = right - dirtyRect.x;
        dirtyRect.height = bottom - dirtyRect.y;
    }
}

function draw() {
    if (dirtyRect) {
        // 只清除脏矩形区域
        ctx.clearRect(dirtyRect.x, dirtyRect.y, dirtyRect.width, dirtyRect.height);
        
        // 只重绘脏矩形区域内的内容
        redrawInRect(dirtyRect);
        
        dirtyRect = null;
    }
    
    requestAnimationFrame(draw);
}

避免不必要的计算 

// 缓存计算结果
const cache = {};

function calculateExpensiveValue(input) {
    if (cache[input]) {
        return cache[input];
    }
    
    const result = performExpensiveCalculation(input);
    cache[input] = result;
    return result;
}

// 预计算常用值
const sinCache = [];
const cosCache = [];
for (let i = 0; i < 360; i++) {
    const rad = i * Math.PI / 180;
    sinCache[i] = Math.sin(rad);
    cosCache[i] = Math.cos(rad);
}

function fastSin(degrees) {
    return sinCache[Math.round(degrees) % 360];
}

function fastCos(degrees) {
    return cosCache[Math.round(degrees) % 360];
}

// 避免内存泄漏
function createParticleSystem() {
    // 使用对象池模式
    const particlePool = [];
    
    function getParticle() {
        if (particlePool.length > 0) {
            return particlePool.pop();
        }
        return new Particle();
    }
    
    function returnParticle(particle) {
        particle.reset();
        particlePool.push(particle);
    }
    
    return {
        getParticle,
        returnParticle
    };
}

性能监控 

let frameCount = 0;
let lastTime = performance.now();
let fps = 0;

function calculateFPS() {
    frameCount++;
    const currentTime = performance.now();
    
    if (currentTime >= lastTime + 1000) {
        fps = Math.round((frameCount * 1000) / (currentTime - lastTime));
        frameCount = 0;
        lastTime = currentTime;
        
        console.log(`FPS: ${fps}`);
    }
}

function draw() {
    calculateFPS();
    
    // 绘制逻辑...
    
    requestAnimationFrame(draw);
}
