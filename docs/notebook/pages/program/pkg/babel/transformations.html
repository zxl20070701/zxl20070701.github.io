AST 转换实战

什么是AST转换

AST转换是指通过修改抽象语法树（AST）来改变源代码的结构和行为。这是Babel插件开发的核心技术。

基本转换流程

  1. 解析：使用@babel/parser将源代码解析为AST
  2. 转换：遍历和修改AST节点
  3. 生成：使用@babel/generator将修改后的AST转换回代码

常见转换模式

节点替换 

const parser = require('@babel/parser');
const traverse = require('@babel/traverse').default;
const t = require('@babel/types');

const code = 'const x = 1 + 2;';
const ast = parser.parse(code);

traverse(ast, {
  BinaryExpression(path) {
    // 将 + 替换为 *
    if (path.node.operator === '+') {
      path.replaceWith(
        t.binaryExpression('*', path.node.left, path.node.right)
      );
    }
  }
});

节点删除 

traverse(ast, {
  VariableDeclarator(path) {
    // 删除名为 'temp' 的变量声明
    if (path.node.id.name === 'temp') {
      path.remove();
    }
  }
});

节点插入 

traverse(ast, {
  Program(path) {
    // 在程序开头插入console.log
    const logStatement = t.expressionStatement(
      t.callExpression(
        t.memberExpression(t.identifier('console'), t.identifier('log')),
        [t.stringLiteral('程序开始执行')]
      )
    );
    path.unshiftContainer('body', logStatement);
  }
});

实用转换示例

箭头函数转普通函数 

const parser = require('@babel/parser');
const traverse = require('@babel/traverse').default;
const t = require('@babel/types');
const generate = require('@babel/generator').default;

const code = 'const add = (a, b) => a + b;';
const ast = parser.parse(code);

traverse(ast, {
  ArrowFunctionExpression(path) {
    // 将箭头函数转换为普通函数
    const functionExpression = t.functionExpression(
      null,
      path.node.params,
      t.blockStatement([
        t.returnStatement(path.node.body)
      ])
    );
    path.replaceWith(functionExpression);
  }
});

const output = generate(ast);
console.log(output.code);

模板字符串转字符串拼接 

traverse(ast, {
  TemplateLiteral(path) {
    // 将模板字符串转换为字符串拼接
    let result = path.node.quasis[0];
    
    for (let i = 1; i < path.node.quasis.length; i++) {
      const expr = path.node.expressions[i - 1];
      const quasi = path.node.quasis[i];
      
      result = t.binaryExpression(
        '+',
        t.binaryExpression('+', result, expr),
        quasi
      );
    }
    
    path.replaceWith(result);
  }
});

移除console语句 

traverse(ast, {
  CallExpression(path) {
    // 移除console.log、console.warn等语句
    if (
      path.node.callee.type === 'MemberExpression' &&
      path.node.callee.object.name === 'console'
    ) {
      path.remove();
    }
  }
});

高级转换技巧

作用域处理 

traverse(ast, {
  FunctionDeclaration(path) {
    // 在函数内部创建唯一变量名
    const uniqueId = path.scope.generateUidIdentifier('temp');
    
    // 在函数开头插入变量声明
    const declaration = t.variableDeclaration('const', [
      t.variableDeclarator(uniqueId, t.numericLiteral(42))
    ]);
    
    path.get('body').unshiftContainer('body', declaration);
  }
});

条件转换 

traverse(ast, {
  IfStatement(path) {
    // 简化 if (true) { ... } 语句
    if (path.node.test.type === 'BooleanLiteral' && path.node.test.value === true) {
      // 替换为if语句的内容
      path.replaceWithMultiple(path.node.consequent.body);
    }
  }
});
