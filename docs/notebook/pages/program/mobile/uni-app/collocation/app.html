app.vue 应用配置

应用生命周期 

<script>
export default {
  onLaunch: function() {
    console.log('App Launch')
    // 应用初始化完成时触发（全局只触发一次）
  },
  onShow: function() {
    console.log('App Show')
    // 应用启动，或从后台进入前台显示时触发
  },
  onHide: function() {
    console.log('App Hide')
    // 应用从前台进入后台时触发
  },
  onError: function(err) {
    console.log('App Error', err)
    // 应用报错时触发
  },
  onUniNViewMessage: function(e) {
    console.log('App onUniNViewMessage', e)
    // 对 nvue 页面发送的数据进行监听
  },
  // 入口组件不需要实现 render 函数，即使实现了也会被 taro 所覆盖
}
</script>

全局样式

在 app.vue 中定义的样式为全局样式，作用于每一个页面。 

/* 全局样式示例 */
page {
  background-color: #f5f5f5;
  font-size: 16px;
}

.container {
  padding: 20px;
  box-sizing: border-box;
}

.text-primary {
  color: #007aff;
}

.text-center {
  text-align: center;
}
